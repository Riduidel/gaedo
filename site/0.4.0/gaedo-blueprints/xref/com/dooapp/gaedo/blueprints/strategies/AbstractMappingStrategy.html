98  	public void generateIdFor(AbstractBluePrintsBackedFinderService<?, DataType, ?> service, DataType toCreate) {
99  		IdGenerator generator = null;
100 		Class<?> objectType = Utils.maybeObjectify(idProperty.getType());
101 		if (Long.class.isAssignableFrom(objectType)) {
102 			generator = new LongGenerator(service, idProperty);
103 		} else if (Integer.class.isAssignableFrom(objectType)) {
104 			generator = new IntegerGenerator(service, idProperty);
105 		} else if (String.class.isAssignableFrom(objectType)) {
106 			generator = new StringGenerator(service, idProperty);
107 		} else {
108 			throw new UnsupportedIdTypeException(objectType + " can't be used as id : we don't know how to generate its values !");
109 		}
110 		generator.generateIdFor(toCreate);
111 	}
