97  	public Vertex loadVertexFor(String objectVertexId, String className) {
98  		CloseableSequence<Vertex> matching = database.getIndex(Index.VERTICES, Vertex.class).get(Properties.value.name(), objectVertexId);
99  		if(matching.hasNext()) {
100 			while(matching.hasNext()) {
101 				Vertex vertex = matching.next();
102 				String vertexTypeName = getEffectiveType(vertex);
103 				if(Kind.literal==GraphUtils.getKindOf(vertex)) {
104 					if(className.equals(vertexTypeName)) {
105 						return vertex;
106 					}
107 				} else {
108 					return vertex;
109 				}
110 			}
111 		}
112 		return null;
113 	}

121 	 * When creating an empty vertex, we immediatly link it to its associated type vertex : a long will as a consequence be linked to the Long class
122 	 * @param vertexId
123 	 * @param valueClass
124 	 * @return
125 	 * @see com.dooapp.gaedo.blueprints.AbstractBluePrintsBackedFinderService#createEmptyVertex(java.lang.String, java.lang.Class)
126 	 */
127 	@Override
128 	protected Vertex createEmptyVertex(String vertexId, Class<? extends Object> valueClass) {
129 		// technical vertex id is no more used by gaedo which only rley upon the getIdOfVertex method !
130 		Vertex returned = database.addVertex(valueClass.getName()+":"+vertexId);
131 		returned.setProperty(Properties.value.name(), vertexId);
132 		if(Literals.containsKey(valueClass)) {
133 			// some literals aren't so ... literal, as they can accept incoming connections (like classes)
134 			returned.setProperty(Properties.kind.name(), Literals.get(valueClass).getKind().name());
135 			returned.setProperty(Properties.type.name(), TypeUtils.getType(valueClass));
136 		} else {
137 			if(repository.containsKey(valueClass)){
138 				returned.setProperty(Properties.kind.name(), Kind.uri.name());
139 			} else if(Tuples.containsKey(valueClass)) {
140 				// some literals aren't so ... literal, as they can accept incoming connections (like classes)
141 				returned.setProperty(Properties.kind.name(), Tuples.get(valueClass).getKind().name());
142 			}
143 			// obtain vertex for type
144 			Vertex classVertex = classTransformer.getVertexFor(getDriver(), valueClass);
145 			Edge toType = getDriver().createEdgeFor(returned, classVertex, TypeProperty.INSTANCE);
146 			/*
147 			 * Make sure literals are literals by changing that particular edge context to a null value.
148 			 *  Notice we COULD have stored literal type as a property, instead of using
149 			 */
150 			toType.setProperty(GraphSail.CONTEXT_PROP, GraphUtils.asSailProperty(GraphUtils.GAEDO_CONTEXT));
151 		}
152 		// Yup, this if has no default else statement, and that's normal.
153 		
154 		return returned;
155 	}

172 	public Edge createEdgeFor(Vertex fromVertex, Vertex toVertex, Property property) {
173 		String edgeNameFor = GraphUtils.getEdgeNameFor(property);
174 		Edge edge = database.addEdge(getEdgeId(fromVertex, toVertex, property), fromVertex, toVertex, edgeNameFor);
175 		String predicateProperty = GraphUtils.asSailProperty(GraphUtils.getEdgeNameFor(property));
176 		edge.setProperty(GraphSail.PREDICATE_PROP, predicateProperty);
177 		Collection<String> contexts = getLens();
178 		StringBuilder contextPropertyBuilder = new StringBuilder();
179 		if(contexts.size()==0) {
180 			contextPropertyBuilder.append(GraphUtils.asSailProperty(GraphSail.NULL_CONTEXT_NATIVE));
181 		} else {
182 			for(String context : contexts) {
183 				if(contextPropertyBuilder.length()>0)
184 					contextPropertyBuilder.append(" " );
185 				contextPropertyBuilder.append(GraphUtils.asSailProperty(context));
186 				
187 			}
188 		}
189 		String contextProperty = contextPropertyBuilder.toString();
190 		edge.setProperty(GraphSail.CONTEXT_PROP, contextProperty);
191 		// Finally build the context-predicate property by concatenating both
192 		edge.setProperty(GraphSail.CONTEXT_PROP + GraphSail.PREDICATE_PROP, contextProperty+" "+predicateProperty);
193 		return edge;
194 	}
