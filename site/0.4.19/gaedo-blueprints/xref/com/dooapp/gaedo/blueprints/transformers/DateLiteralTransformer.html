14  /**
15   * As SimpleDateFormat is not thread-safe and construction heavy, see here
16   * (http:
17   * //www.jmdoudoux.fr/java/dej/chap-utilisation_dates.htm#utilisation_dates-3)
18   * for the why of this pattern
19   * 
20   * Furthermore, used pattern is ISO standard (see there : http://www.fileformat.info/tip/java/simpledateformat.htm)
21   * 
22   * @author ndx
23   * 
24   */
25  public class DateLiteralTransformer extends AbstractSimpleLiteralTransformer<Date> implements LiteralTransformer<Date> {
26  	
27  	/**
28  	 * Map linking date format specification (a uri like xsd:date or http://www.w3.org/2001/XMLSchema#date) to the format string used to parse it.
29  	 */
30  	private static final Map<String, String> FORMATS = new TreeMap<String, String>();
31  	
32  	static {
33  		FORMATS.put("xsd:date", "yyyy-MM-dd'T'HH:mm:ssz");
34  		FORMATS.put("http://www.w3.org/2001/XMLSchema#date", "yyy-MM-dd");
35  	}
36  					
37  	
38  	/**
39  	 * A map allowing lazy loading of those inconvenient threaded loaders.
40  	 */
41  	private static final Map<String, DateFormatThreadedLoader> loaders = new TreeMap<String, DateFormatThreadedLoader>();
42  	
43  	/**
44  	 * Cache linking dates to toeir associated format definition.
45  	 * As the dates are in memory, and the formats are constants, this weak hash map should have very low memory impact.
46  	 */
47  	private static Map<Date, String> dateCache = new WeakHashMap<Date, String>();
48  	
49  	public static DateFormatThreadedLoader getLoader(String format) {
50  		if(!loaders.containsKey(format))
51  			loaders.put(format, new DateFormatThreadedLoader(format));
52  		return loaders.get(format);
53  	}
54  	
55  	public DateLiteralTransformer() {
56  		super(Date.class);
57  	}
58  
59  	@Override
60  	protected Object getVertexValue(Date value) {
61  		// fallback value
62  		String format = getTypeOf(value);
63  		return getLoader(FORMATS.get(format)).format(value);
64  	}
65  
66  	public Date loadObject(GraphDatabaseDriver driver, Class valueClass, Vertex key) {
67  		String property = driver.getValue(key).toString();
68  		String type = key.getProperty(Properties.type.name()).toString();
69  		try {
70  			Date returned = getLoader(FORMATS.get(type)).parse(property);
71  			dateCache.put(returned, type);
72  			return returned;
73  		} catch (ParseException e) {
74  			throw new BadLiteralException("\"" + property + "\" can't be efficiently parsed from a \""+type+"\" supposed to contain a date");
75  		}
76  	}
77  	
78  	/**
79  	 * Type of date depends upon the date
80  	 * @param value
81  	 * @return
82  	 * @see com.dooapp.gaedo.blueprints.transformers.AbstractLiteralTransformer#getTypeOf(java.lang.Object)
83  	 */
84  	@Override
85  	public String getTypeOf(Object value) {
86  		String format = "xsd:date";
87  		if(dateCache.containsKey(value)) {
88  			format = dateCache.get(value);
89  		}
90  		dateCache.put((Date) value, format);
91  		return format;
92  	}
