90  	/**
91  	 * Get optimal query for finding results as fast as possible
92  	 * 
93  	 * @param datastore
94  	 *            used datastore
95  	 * @param sortingExpression
96  	 *            sorting expression used to organize results
97  	 * @return one of {@link ClassicalQuery} or ??? depending upon the best
98  	 *         executor for query
99  	 */
100 	public DataStoreExecutableQuery getQuery(DatastoreService datastore,
101 			SortingExpression sortingExpression) {
102 		boolean onlyOnKeys = true;
103 		Collection<Key> keys = new LinkedList<Key>();
104 		for (FilterPredicate predicate : query.getFilterPredicates()) {
105 			onlyOnKeys &= (KEY_PROPERTY.equals(predicate.getPropertyName()) && Query.FilterOperator.EQUAL
106 					.equals(predicate.getOperator()));
107 			if (onlyOnKeys)
108 				keys.add((Key) predicate.getValue());
109 		}
110 		if (onlyOnKeys) {
111 			// If there is a non-empty sorting expression on this key query,
112 			// there is a problem
113 			if (sortingExpression != null
114 					&& sortingExpression.iterator().hasNext()) {
115 				logger.log(Level.WARNING,
116 						"unable to sort queries based on key only",
117 						new Exception());
118 			}
119 			return new KeysQueryLookup(datastore, keys, expectedKind);
120 		}
121 		if (sortingExpression != null) {
122 			for (Entry<FieldInformer, SortingExpression.Direction> entry : sortingExpression) {
123 				query
124 						.addSort(
125 								Utils.getDatastoreFieldName(entry.getKey()
126 										.getField()),
127 								entry.getValue() == Direction.Ascending ? SortDirection.ASCENDING
128 										: SortDirection.DESCENDING);
129 			}
130 		}
131 		return new ClassicalQuery(datastore, query);
132 	}

278 	/*
279 	 * Remove generic types to fix strange compilation failure with maven:
280 	 * gaedo/gaedo-google-datastore/src
281 	 * /main/java/org/gaedo/google/datastore/GAEQueryBuilder.java:[199,87]
282 	 * inconvertible types found :
283 	 * com.dooapp.gaedo.finders.FinderCrudService<capture#345 of

285 	 * org.gaedo.google.datastore.DatastoreFinderService<?,?>

287 	@SuppressWarnings("unchecked")
288 	private Object resolve(Property field, Object value) {
289 		if (idManager.isIdField(field))
290 			return idManager.buildKey(query.getKind(), value);
291 		try {
292 			DatastoreFinderService service = (DatastoreFinderService) repository
293 					.get(field.getGenericType());
294 			return service.getIdManager().getKey(service.getKind(), value);
295 		} catch (NoSuchServiceException e) {
296 			// Simple data, return it
297 			return value;
298 		}
299 	}
