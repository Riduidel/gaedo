22  public class ScanningPropertiesBuilder extends VoidVisitorAdapter<InformerInfos> {
23  	private boolean includeStaticProperties;
24  	/**
25  	 * Elements from this collection can be in two forms
26  	 * <ul>
27  	 * <li>Simple strings</li>
28  	 * <li>Javadoc-comaptible properties : package.class#property</li>
29  	 * </ul>
30  	 */
31  	private Collection<String> excludedProperties;

33  	public ScanningPropertiesBuilder(boolean includeStaticProperties, Collection<String> excludedProperties) {
34  		this.includeStaticProperties = includeStaticProperties;
35  		this.excludedProperties = excludedProperties;
36  	}

40  		arg.imports.add(n);
41  	}

48  	@Override
49  	public void visit(ClassOrInterfaceDeclaration classDeclaration, InformerInfos informer) {
50  		if (!classDeclaration.isInterface()) {
51  			informer.className = classDeclaration.getName();
52  			if (classDeclaration.getExtends()!=null && classDeclaration.getExtends().size() > 0) {
53  				informer.superClassName = classDeclaration.getExtends().get(0).getName();
54  			}
55  			informer.properties = filter(createPropertiesFrom(classDeclaration.getMembers()), informer);
56  		}
57  	}

59  	/**
60  	 * Filter map of properties using excluded properties
61  	 * @param createPropertiesFrom
62  	 * @param informer TODO
63  	 * @return
64  	 */
65  	private Collection<PropertyInfos> filter(Map<String, PropertyInfos> createPropertiesFrom, InformerInfos informer) {
66  		Collection<String> removed = new LinkedList<String>(); 
67  		String qualifiedClassName = informer.classPackage+"."+informer.className;
68  		// Now exclude properties based upon their name and their matching
69  		for(String s : excludedProperties) {
70  			int dashIndex = s.indexOf('#');
71  			if(dashIndex>0) {
72  				if(s.substring(0, dashIndex).equals(qualifiedClassName)) {
73  					removed.add(s.substring(dashIndex+1));
74  				}
75  			} else {
76  				// a trick : if map contains property name, it will be removed. Otherwise nothing will happen
77  				removed.add(s);
78  			}
79  		}
80  		for(String s : removed) {
81  			createPropertiesFrom.remove(s);
82  		}
83  		return createPropertiesFrom.values();
84  	}

86  	/**
87  	 * Create a list of properties from a class body declaration
88  	 * @param members list of class members (fields and methods)
89  	 * @return a map linking property name to its value
90  	 */
91  	private Map<String, PropertyInfos> createPropertiesFrom(List<BodyDeclaration> members) {
92  		Map<String, PropertyInfos> properties = new TreeMap<String, PropertyInfos>();
93  		for (BodyDeclaration b : members) {
94  			if (b instanceof FieldDeclaration) {
95  				FieldDeclaration f = (FieldDeclaration) b;
96  				if(includeStaticProperties || !Modifier.isStatic(f.getModifiers())) {
97  					for (VariableDeclarator d : f.getVariables()) {
98  						Type type = f.getType();
99  						String name = d.getId().getName();
100 						addInfosFor(properties, InfosTypes.FIELD, name, type);
101 					}
102 				}
103 			} else if (b instanceof MethodDeclaration) {
104 				MethodDeclaration m = (MethodDeclaration) b;
105 				if(includeStaticProperties || !Modifier.isStatic(m.getModifiers())) {
106 					if (isGetter(m)) {
107 						addInfosFor(properties, InfosTypes.GETTER, m.getName(), m.getType());
108 					} else if (isSetter(m)) {
109 						addInfosFor(properties, InfosTypes.SETTER, m.getName(), m.getParameters().get(0).getType());
110 					}
111 				}
112 			}
113 		}
114 		return properties;
115 	}

117 	private void addInfosFor(Map<String, PropertyInfos> properties, InfosTypes field, String name, Type type) {
118 		String propertyName = field.getNameFor(name);
119 		if (!properties.containsKey(propertyName)) {
120 			properties.put(propertyName, new PropertyInfos());
121 		}
122 		PropertyInfos propertyInfos = properties.get(propertyName);
123 		propertyInfos.name = propertyName;
124 		propertyInfos.type = type;
125 		switch (field) {
126 		case FIELD:
127 			propertyInfos.field = name;
128 			break;
129 		case GETTER:
130 			propertyInfos.getter = name;
131 			break;
132 		case SETTER:
133 			propertyInfos.setter = name;
134 			break;
135 		}
136 	}

138 	private boolean isSetter(MethodDeclaration m) {
139 		if (m.getName().startsWith(InfosTypes.SETTER.prefix)) {
140 			return m.getType().toString().equals("void");
141 		}
142 		return false;
143 	}
144 
145 	private boolean isGetter(MethodDeclaration m) {
146 		if (m.getName().startsWith(InfosTypes.GETTER.prefix)) {
147 			return m.getParameters()==null || m.getParameters().size() == 0;
148 		}
149 		return false;
150 	}
