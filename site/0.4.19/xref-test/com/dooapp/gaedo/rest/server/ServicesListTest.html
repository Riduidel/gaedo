28  	public void testServicesListInXml() throws Exception {
29  		final Collection<Class<? extends Serializable>> containedClasses = new LinkedList<Class<? extends Serializable>>(Arrays.asList(User.class, Post.class));
30  		ClientResource serviceList = new ClientResource("http://localhost:"+TEST_PORT+TEST_DIR+GaedoResourceApplication.SERVICE_LIST_ROUTE);
31  		Representation representation = serviceList.get(MediaType.TEXT_XML);
32  		Assert.assertThat(representation.getMediaType(), Is.is(MediaType.TEXT_XML));
33  		NodeTest nodeTest = new NodeTest(representation.getReader());
34  		NodeTester nodeTester = new AbstractNodeTester() {
35  			@Override
36  			public void testElement(Element element) throws NodeTestException {
37  				String nodeName = element.getNodeName();
38  				if(ServicesList.SERVICE_LIST_NODE_NAME.equals(nodeName)) {
39  					if(element.getAttributes().getLength()>0) {
40  						throw new NodeTestException("element "+ServicesList.SERVICE_LIST_NODE_NAME+" is not expected to have any attribute");
41  					}
42  				} else if(ServicesList.SERVICE_NODE_NAME.equals(nodeName)) {
43  					String dataType = element.getAttribute(ServicesList.DATA_TYPE_ATTRIBUTE);
44  					/*
45  					 * Unchecked code is used here to simplify some rather complex calls
46  					 */
47  					try {
48  						Class containedType = Class.forName(dataType);
49  						containedClasses.remove(containedType);
50  						FinderCrudService service = repository.get(containedType);
51  						Class serviceInterface = Class.forName(element.getAttribute(ServicesList.SERVICE_TYPE_ATTRIBUTE));
52  						if(!serviceInterface.isInstance(service)) {
53  							throw new NodeTestException("service "+service+" is not an instance of interface "+serviceInterface.getName()+" which designates it", element);
54  						}
55  						Class informerType = Class.forName(element.getAttribute(ServicesList.INFORMER_TYPE_ATTRIBUTE));
56  						if(!informerType.isInstance(service.getInformer())) {
57  							throw new NodeTestException("service informer "+service.getInformer()+" is not an instance of interface "+informerType.getName()+" which designates it", element);
58  						}
59  					} catch(NodeTestException e) {
60  						throw e;
61  					} catch(Exception e) {
62  						e.printStackTrace();
63  						throw new NodeTestException("something went wrong while testing node due to inner exception "+e.getMessage(), element);
64  					}
65  				} else {
66  					throw new NodeTestException("no node name other than "+ServicesList.SERVICE_LIST_NODE_NAME+" and "
67  							+ServicesList.SERVICE_NODE_NAME+" is allowed in this document", element);
68  				}
69  			}
70  		};
71  		nodeTest.performTest(nodeTester, Node.ELEMENT_NODE);
72  		Assert.assertThat(containedClasses.size(), Is.is(0));
73  	}
