31  	/**
32  	 * Effective implementation method.
33  	 * @param annotation TODO
34  	 * @param provider Property provider used to get fields of various classes
35  	 * @param inputClass the class the user wants an id for
36  	 * @param currentClass the currently examined class
37  	 * @param expectedIdClasses possible classes of field containing id
38  	 * @return property holding the id, or a {@link BadIdAnnotatedClassException} is thrown if none is found
39  	 */
40  	private static Property locateOneField(Annotations annotation, PropertyProvider provider, Class<?> inputClass, Class<?> currentClass, Collection<Class> expectedIdClasses) {
41  		Property[] fields = provider.get(currentClass);
42  		List<Property> probableId = AnnotationsFinder.findAll(fields,
43  				Annotations.ID);
44  		Property idField = null;
45  		if (probableId.size() == 1) {
46  			Property potentialIdField = probableId.get(0);
47  			/*
48  			 * Someone using Shcrodinger cat in a Java blog post can't be evil
49  			 * ;-) http://sensualjava.blogspot.com/2007
50  			 * /11/reflection-and-auto-boxing.html
51  			 */
52  			if(expectedIdClasses!=null) {
53  				boolean found = false;
54  				Iterator<Class> iterator = expectedIdClasses.iterator();
55  				while(iterator.hasNext() && !found) {
56  					found = potentialIdField.getType().isAssignableFrom(iterator.next());
57  				}
58  				if(found) {
59  					idField = potentialIdField;
60  				}
61  			}
62  		}
63  		// maybe the parent has the solution
64  		if(idField==null) {
65  			Class<?> nextClass = currentClass.getSuperclass();
66  			if(Object.class.equals(nextClass)) {
67  				throw new BadIdAnnotatedClassException(inputClass, expectedIdClasses);
68  			} else {
69  				idField = locateOneField(annotation, provider, inputClass, nextClass, expectedIdClasses);
70  			}
71  		}
72  		return idField;
73  	}

105 		if(!Object.class.equals(currentClass)) {
106 			Property[] fields = provider.get(currentClass);
107 			List<Property> returned = AnnotationsFinder.findAll(fields,
108 					id);
