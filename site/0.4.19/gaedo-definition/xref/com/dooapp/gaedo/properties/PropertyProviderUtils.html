34  	public static Property[] getAllProperties(PropertyProvider provider, Class<?> containedClass) {
35  		Collection<Property> temporary = new LinkedList<Property>();
36  		Class<?> currentClass = containedClass;
37  		while(!Object.class.equals(currentClass)) {
38  			temporary.addAll(Arrays.asList(provider.get(currentClass)));
39  			currentClass = currentClass.getSuperclass();
40  		}
41  		return temporary.toArray(new Property[temporary.size()]);
42  	}
