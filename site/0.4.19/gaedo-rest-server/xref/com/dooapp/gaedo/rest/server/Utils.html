35  	 * Split string according to splitter and removes empty matches
36  	 * @param key
37  	 * @return
38  	 */
39  	public static String[] split(String key) {
40  		String[] splitted = splitter.split(key, -1);
41  		Collection<String> returned = new LinkedList<String>();
42  		for(String s : splitted) {
43  			if(s.length()>0) {
44  				returned.add(s);
45  			}
46  		}
47  		return returned.toArray(new String[returned.size()]);
48  	}

50  	/**
51  	 * Transform the map of key to values into a tree of key to values.
52  	 * @param paramsMap a map of parameters in the form "prefix[dimension_1][dimension_2]...[dimension_n]=somerandomvalue" 
53  	 * @return a map of maps of maps in the form
54  	 * <ul>
55  	 * 			<li>dimension_1
56  	 * 				<ul>
57  	 * 					<li>dimension_2 (and so on)</li>
58  	 * 				</ul>
59  	 * 			</li>
60  	 * </ul>
61  	 * Notice that prefix is totally consumed here
62  	 */
63  	public static Map<String, Object> getValuesAsTree(Map<String, Object> paramsMap) {
64  		Map<String, Object> returned = new LinkedHashMap<String, Object>();
65  		for(Map.Entry<String, Object> param : paramsMap.entrySet()) {
66  			String[] paramChain = split(param.getKey());
67  			Map<String, Object> current = returned;
68  			for (int index = 0; index < paramChain.length; index++) {
69  				if(index<paramChain.length-1) {
70  					if(!current.containsKey(paramChain[index])) {
71  						current.put(paramChain[index], new LinkedHashMap<String, Object>());
72  					}
73  					current = (Map<String, Object>) current.get(paramChain[index]);
74  				} else {
75  					current.put(paramChain[index], param.getValue());
76  				}
77  			}
78  		}
79  		return returned;
80  	}
