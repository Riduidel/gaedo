52  	/**
53  	 * Get informer for given type, by reflective upon its attribute
54  	 * @param property source property
55  	 * @param genericType generic type of property
56  	 * @return
57  	 */
58  	protected FieldInformer getInformerFor(Property property, Type genericType) {
59  		if (genericType instanceof Class) {
60  			// Any other case is a class one (look at
61  			// http://java.sun.com/j2se/1.5.0/docs/api/java/lang/reflect/Type.html
62  			// for more info)
63  			Class<?> clazz = (Class<?>) genericType;
64  			Class<? extends FieldInformer> informerClass = getInformerForClass(clazz);
65  			// This class must have a constructor using one Property parameter, elsewhere mayhem may come
66  			Constructor<? extends FieldInformer> used;
67  			try {
68  				used = informerClass.getDeclaredConstructor(Property.class);
69  				return used.newInstance(property);
70  			} catch (Exception e) {
71  				throw new UnableToCreateFieldInformerException(clazz, informerClass, property.getClass(), e);
72  			}
73  		} else if(genericType instanceof ParameterizedType) {
74  			// Too lazy to write it all
75  			ParameterizedType pType = (ParameterizedType) genericType;
76  			return getInformerFor(property, pType.getRawType());
77  		}
78  		return null;
79  	}

81  	/**
82  	 * Helper method allowing somee overriding of class localization
83  	 * @param clazz
84  	 * @return
85  	 */
86  	protected Class<? extends FieldInformer> getInformerForClass(Class<?> clazz) {
87  		Class<? extends FieldInformer> existingOne = hierarchicalySearchForInformer(clazz);
88  		if(existingOne==null)
89  			throw new NoInformerForThisTypeException(clazz);
90  		else
91  			return existingOne;
92  	}

94  	/**
95  	 * Search in all superclasses of given class for informers
96  	 * @param clazz source class
97  	 * @return informer, if found, null elsewhen.
98  	 */
99  	private Class<? extends FieldInformer> hierarchicalySearchForInformer(Class<?> clazz) {
100 		Class<? extends FieldInformer> existingOne = null;
101 		if(existingInformerMappings.containsKey(clazz)) {
102 			return existingInformerMappings.get(clazz);
103 		} else {
104 			Map<Class<?>, Class<? extends FieldInformer>> foundInformers = new HashMap<Class<?>, Class<? extends FieldInformer>>();
105 			for(Class<?> interfazz : clazz.getInterfaces()) {
106 				if(existingInformerMappings.containsKey(clazz))
107 					foundInformers.put(interfazz, existingInformerMappings.get(clazz));
108 			}
109 			if(foundInformers.size()==1) {
110 				return foundInformers.entrySet().iterator().next().getValue();
111 			} else if(foundInformers.size()>1) {
112 				throw new ConflictingInformersForThisTypeException(clazz, foundInformers);
113 			} else if(Object.class.equals(clazz)) {
114 				return null;
115 			} else {
116 				if(clazz.getSuperclass()!=null) {
117 					return hierarchicalySearchForInformer(clazz.getSuperclass());
118 				} else {
119 					return null;
120 				}
121 			}
122 		}
123 	}
