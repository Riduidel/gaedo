47  	 * Generate an instance of one of the concrete types corresponding to input
48  	 * type (if input type is not cocnrete)
49  	 * 
50  	 * @param rawContainerType
51  	 * @param property
52  	 * @return
53  	 */
54  	@SuppressWarnings("unchecked")
55  	public static <DataType> Collection<DataType> generateCollection(Class<?> rawContainerType,
56  			Collection<DataType> property) {
57  		if (property == null) {
58  			if (rawContainerType.isInterface()) {
59  				if (BlockingQueue.class.isAssignableFrom(rawContainerType)) {
60  					property = new ArrayBlockingQueue<DataType>(10);
61  				} else if (Queue.class.isAssignableFrom(rawContainerType)) {
62  					property = new ArrayBlockingQueue<DataType>(10);
63  				} else if (NavigableSet.class.isAssignableFrom(rawContainerType)) {
64  					property = new TreeSet<DataType>();
65  				} else if (Set.class.isAssignableFrom(rawContainerType)) {
66  					property = new HashSet<DataType>();
67  				} else if (SortedSet.class.isAssignableFrom(rawContainerType)) {
68  					property = new TreeSet<DataType>();
69  				} else if (List.class.isAssignableFrom(rawContainerType)) {
70  					property = new LinkedList<DataType>();
71  				} else if (Collection.class.isAssignableFrom(rawContainerType)) {
72  					property = new LinkedList<DataType>();
73  				}
74  			} else if (Modifier.isAbstract(rawContainerType.getModifiers())) {
75  				throw new UnsupportedOperationException(
76  						"are you kiddin or what ? Replace your abstract type "
77  								+ rawContainerType.getName()
78  								+ "by an interface, it's wayyyy better !");
79  			} else {
80  				try {
81  					property = (Collection<DataType>) rawContainerType.newInstance();
82  				} catch (Exception e) {
83  					throw new UnableToCreateObjectException(e, rawContainerType);
84  				}
85  			}
86  		}
87  		return property;
88  	}

236 	public static Map<String, Method> getUppercasedMap(Method[] methods) {
237 		Map<String, Method> fieldNames = new HashMap<String, Method>();
238 		for(Method method : methods) {
239 			if(!method.getDeclaringClass().equals(Object.class)) {
240 				String name = method.getName();

251 	/**
252 	 * Convert a string into an object of the class it is supposed to come from.
253 	 * This method is expected to work only for String (quite logical, isn't ?) and classes providing either a constructor using String as an argument,
254 	 * or a valueOf method having String as an argument. Any other case will miserably fail (ie a NoFromStringConversionExists will be thrown)
255 	 * @param value input string value
256 	 * @param type expected output type
257 	 * @return an object of that type
258 	 */
259 	public static <Type> Type fromString(String value, Class<Type> type) {
260 		if(String.class.equals(type)) {
261 			return (Type) value.toString();
262 		} else if(type.isPrimitive()) {
263 			return (Type) fromString(value, objectify(type));
264 		} else if(URI.class.equals(type)) {
265 			try {
266 				return (Type) new URI(value);
267 			} catch(URISyntaxException e) {
268 				throw new UnableToBuilddURIException("\""+value+"\" can't be transformed into an URI object", e);
269 			}
270 		} else if(Class.class.equals(type)) {
271 			return classFromString(value);
272 		} else {
273 			/* First check if a constructor exists */
274 			try {
275 				Constructor<Type> withString = type.getDeclaredConstructor(String.class);
276 				return withString.newInstance(value);
277 			} catch(Exception e) {
278 				/* This constructor does not seems to exists. Is there any chance a "valueOf" method exists (useful for numbers objects) ? */
279 				try {
280 					Method valueOf = type.getDeclaredMethod("valueOf", String.class);
281 					return (Type) valueOf.invoke(null, value);
282 				} catch (Exception e1) {
283 					/* Seems like the reply is no */
284 					throw new NoFromStringConversionExistsException(type, e, e1);
285 				}
286 			}
287 		}
288 	}

290 	/**
291 	 * Try to load given type class
292 	 * @param value
293 	 * @return
294 	 */
295 	private static <Type> Type classFromString(String value) {
296 		value = maybeObjectify(value);
297 		ClassLoader[] used = new ClassLoader[] {value.getClass().getClassLoader(), Utils.class.getClassLoader(), Thread.currentThread().getContextClassLoader()};
298 		for(ClassLoader c : used) {
299 			if(c!=null) {
300 				try {
301 					return (Type) c.loadClass(value);
302 				} catch (ClassNotFoundException e) {
303 					// nothing to do, a better exception will be built later
304 				}
305 			}
306 		}
307 		throw new UnableToLoadClassException(value, used);
308 	}

333 		if(declaring!=null) {
334 			if(!declaring.equals(Object.class)) {
335 				returned.add(declaring);
