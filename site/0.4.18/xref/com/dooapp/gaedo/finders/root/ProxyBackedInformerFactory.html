9   /**
10   * This time come the authentical rock'n'roll ! This class will implement
11   * {@link Informer} objects from sub-interfaces by redirecting calls to existing
12   * {@link Informer#get(String)} method and casting the result to the correct
13   * type .
14   * 
15   * As an example, suppose you have a classic User bean (this is only an example
16   * implementation, that I wouldn't be proud letting go out of my javadoc):
17   * 
18   * <pre>
19   * public class User {
20   * 		private String login;
21   * 		private String password
22   * }
23   * </pre>
24   * 
25   * One theorically have to write the full Informer for that bean, which is
26   * tedious and absolutely without the slightest interest (read boilerplate code
27   * or why can't you generate it).
28   * 
29   * Using this class, it is possible to create only an informer interface
30   * (declaring the methods to use, and gaedo will guess their code for you.
31   * 
32   * As a consequence, suppose you want your informer to look like
33   * 
34   * <pre>
35   * public interface UserInformer extends Informer&lt;User&gt; {
36   * 	public StringInformer getLogin();
37   * 
38   * 	public StringInformer getPassword();
39   * }
40   * </pre>
41   * 
42   * All you have to do is call
43   * 
44   * <pre>
45   * ProxyBackedInformerFactory factory = new ProxyBackedInformerFactory(/ *the informer factory to use* /)
46   * UserInformer usable = factory.get(UserInformer.class, User.class);
47   * </pre>
48   * 
49   * And gaedo will do all that boilerplate code for you.
50   * 
51   * @author Nicolas
52   */
53  public class ProxyBackedInformerFactory extends AbstractInformerFactory
54  		implements InformerFactory {
55  	public ProxyBackedInformerFactory(
56  			ReflectionBackedInformerFactory reflectiveInformerFactory) {
57  		super(reflectiveInformerFactory);
58  	}

60  	/**
61  	 * Build invocation handler and the rest of the required code for the input
62  	 * interface
63  	 * 
64  	 * @param <InformerType>
65  	 * @param <ContainedType>
66  	 * @param informerClass
67  	 *            expected informer class to implement
68  	 * @param containedType
69  	 *            contained object type
70  	 * @return an invocation handler that will transform calls to correct ones.
71  	 */
72  	protected <InformerType extends Informer<ContainedType>, ContainedType> InformerType createDynamicInformer(
73  			Class<InformerType> informerClass,
74  			Class<ContainedType> containedType,
75  			Informer<ContainedType> existingInformer) {
76  		ClassLoader classLoader = informerClass.getClassLoader();
77  		Class<?>[] interfaces = new Class<?>[] { informerClass, Proxied.class, Validable.class };
78  		InformerClassInvocationHandler invocationHandler = new InformerClassInvocationHandler(informerClass, containedType, existingInformer);
79  		return (InformerType) Proxy.newProxyInstance(
80  					classLoader, 
81  					interfaces, 
82  					invocationHandler);
83  	}
