60  	public GraphBasedMappingStrategy(Class<DataType> serviceContainedClass, PropertyProvider propertyProvider, Migrator migrator) {
61  		super(serviceContainedClass, propertyProvider, migrator);
62  		beanPropertiesForServiceClass = StrategyUtils.getBeanPropertiesFor(propertyProvider, serviceContainedClass, migrator);
63  		// an auxiliary list is created to navigate properties, as we will
64  		// remove some of them from beanProperties,
65  		// and that is not something iterators are good at
66  		List<Property> propertyNavigator = new LinkedList<Property>(beanPropertiesForServiceClass.keySet());
67  		// Replace id property by one allowing zero navigation (for findById and
68  		// other methods)
69  		for (Property p : propertyNavigator) {
70  			if (p.getAnnotation(BagProperty.class) != null) {
71  				beanPropertiesForServiceClass.remove(p);
72  			}
73  		}
74

77  	/**
78  	 * To get contained property, we simply build a map of edges outgoing from
79  	 * vertex and associate to them cascade information based on ... what ?
80  	 * magic ?
81  	 * 
82  	 * @param object
83  	 *            object to load data in
84  	 * @param vertex
85  	 *            associated vertew which will never be left null
86  	 * @return mapping between properties representing edges and cascade for
87  	 *         these properties
88  	 * @see com.dooapp.gaedo.blueprints.strategies.GraphMappingStrategy#getContainedProperties(java.lang.Object,
89  	 *      com.tinkerpop.blueprints.pgm.Vertex, CascadeType)
90  	 */
91  	@Override
92  	public Map<Property, Collection<CascadeType>> getContainedProperties(DataType object, Vertex vertex, CascadeType cascadeType) {
93  		// notice that, as default, properties only cascade creation
94  		Map<Property, Collection<CascadeType>> returned = new HashMap<Property, Collection<CascadeType>>();
95  		// During persist operation, we only persist graph properties, and NOT
96  		// object ones (that would polute the graph)
97  		if (!CascadeType.MERGE.equals(cascadeType) && !CascadeType.PERSIST.equals(cascadeType))
98  			returned.putAll(beanPropertiesForServiceClass);
99  		// extract edges from persistable object infos
100 		if (object instanceof PropertyMapPropertyAccess) {
101 			PropertyMapPropertyAccess propertyBag = (PropertyMapPropertyAccess) object;
102 			for (Property uri : propertyBag.propertyUris()) {
103 				returned.put(uri, StrategyUtils.extractCascadeOfJPAAnnotations(uri));
104 			}
105 		}
106 		if (vertex == null) {
107 		} else {
108 			Map<String, GraphBasedPropertyBuilder<DataType>> edgeLabelToProperty = new TreeMap<String, GraphBasedPropertyBuilder<DataType>>();
109 			for (Edge e : vertex.getOutEdges()) {
110 				String edgeLabel = e.getLabel();
111 				if (!edgeLabelToProperty.containsKey(edgeLabel)) {
112 					edgeLabelToProperty.put(edgeLabel, new GraphBasedPropertyBuilder<DataType>(serviceContainedClass, service.getDriver(), service.getLens()));
113 				}
114 				edgeLabelToProperty.get(edgeLabel).add(e);
115 			}
116 			for (GraphBasedPropertyBuilder<DataType> builder : edgeLabelToProperty.values()) {
117 				try {
118 					Property built = builder.build();
119 					// only add property if absent from properties laoded from the bean, as properties loaded from the bean are respectfull to initial data type
120 					if(!returned.containsKey(built))
121 						returned.put(built, StrategyUtils.extractCascadeOfJPAAnnotations(built));
122 				} catch(NoEdgeInNamedGraphsException e) {
123 					logger.info(e.getMessage());
124 				}
125 			}
126 		}
127 		return returned;
128 	}

130 	/**
131 	 * When loading that strategy, we also customize the informer factory to
132 	 * make sure non-existing properties
133 	 * 
134 	 * @param service
135 	 * @see com.dooapp.gaedo.blueprints.strategies.AbstractMappingStrategy#reloadWith(com.dooapp.gaedo.blueprints.AbstractBluePrintsBackedFinderService)
136 	 */
137 	@Override
138 	public void reloadWith(AbstractBluePrintsBackedFinderService<?, DataType, ?> service) {
139 		super.reloadWith(service);
140 		InformerFactory informerFactory = service.getInformerFactory();
141 		if (informerFactory instanceof ProxyBackedInformerFactory) {
142 			ProxyBackedInformerFactory initialProxy = (ProxyBackedInformerFactory) informerFactory;
143 			ReflectionBackedInformerFactory initialReflective = initialProxy.getReflectiveInformerFactory();
144 			
145 			FieldInformerLocator initialLocator = initialReflective.getFieldLocator();
146 			// nothing has been loaded yet
147 			if(initialLocator instanceof CumulativeFieldInformerLocator) {
148 				ReflectionBackedInformerFactory usedReflective = new ReflectionBackedInformerFactory(
149 								new DelegatingInformerLocator(initialLocator, createLocator(initialLocator)),
150 								initialReflective.getPropertyProvider()
151 								);
152 				
153 				ProxyBackedInformerFactory usedProxy = new ProxyBackedInformerFactory(usedReflective);
154 				service.setInformerFactory(usedProxy);
155 			} else if(initialLocator instanceof DelegatingInformerLocator) {
156 				DelegatingInformerLocator delegating = (DelegatingInformerLocator) initialLocator;
157 				delegating.setSecond(createLocator(delegating.getFirst()));
158 			}
159 		}
160 	}

163 		return new GraphFieldLocator<DataType>(

166 						service.getDatabase(), service
167 						.getDriver(),
168 		/*
169 		 * yup, there is a loop here, but for a good reason :
170 		 * graphFieldLocator will generate a Property out of an edge, then

173 		 */
174 		cumulativeLocator,

221 	/**
222 	 * What are the objects requiring loading ? The ones that are directly accessed.
223 	 * As a consequence, we use a {@link ThreadLocal} mechanism to store objects ids being loaded.
224 	 * In other words, if thread local contains no object id, put current one.
225 	 * If there is one (which is not current one, return false).
226 	 * Notice the {@link #loaded(String, Vertex, Object, Map)} method, when invoked, will clear that info for queries to run smoothly
227 	 * @param objectVertexId
228 	 * @param objectVertex
229 	 * @param objectsBeingAccessed
230 	 * @return
231 	 * @see com.dooapp.gaedo.blueprints.strategies.GraphMappingStrategy#shouldLoadPropertiesOf(java.lang.String, com.tinkerpop.blueprints.pgm.Vertex, java.util.Map)
232 	 */
233 	@Override
234 	public boolean shouldLoadPropertiesOf(String objectVertexId, Vertex objectVertex, Map<String, Object> objectsBeingAccessed) {
235 		if(loadedByThisThread.get()==null) {
236 			loadedByThisThread.set(objectVertexId);
237 		}
238 		return loadedByThisThread.get().equals(objectVertexId);
239 	}
