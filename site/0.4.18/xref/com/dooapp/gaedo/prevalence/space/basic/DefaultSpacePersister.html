76  	 * Logs command to a file named from commandIndex. Notice only Commands
77  	 * implementing the Serializable interface will be logged, as the other ones
78  	 * don't need to be logged.
79  	 * 
80  	 * @param toLog
81  	 *            command to log
82  	 */
83  	@Override
84  	public <CommandReturnType, Key extends Serializable> void logCommand(
85  			Command<CommandReturnType, Key> toLog) {
86  		if (toLog instanceof Serializable) {
87  			// This is a kind of double effect synchronized locking, as all code
88  			// from this class should only be called by a specific executor
89  			// service
90  			synchronized (this) {
91  				writeCommand(config.getCommandIndexAndIncrement(), toLog);
92  			}
93  		}
94  	}

96  	protected <CommandReturnType, Key extends Serializable> void writeCommand(
97  			long commandIndex, Command<CommandReturnType, Key> command) {
98  		File toRead = buildCommandLogFileName(commandIndex);
99  		ObjectOutputStream objectStream = null;
100 		try {
101 			objectStream = new ObjectOutputStream(new FileOutputStream(toRead));
102 			objectStream.writeObject(command);
103 		} catch (IOException e) {
104 			throw new ConsistencyException(e);
105 		} finally {
106 			if (objectStream != null) {
107 				try {
108 					objectStream.close();
109 				} catch (IOException e) {
110 					e.printStackTrace();
111 				}
112 			}
113 		}
114 	}

149 	 * 
150 	 * @return
151 	 */
152 	<Key extends Serializable> Collection<Command<?, Key>> loadCommands(
153 			long low, long high) {
154 		Collection<Command<?, Key>> returned = new LinkedList<Command<?, Key>>();
155 		for (long index = low; index < high; index++) {
156 			if (buildCommandLogFileName(index).exists()) {
157 				// Weird generics bug, no ? Will solve it later
158 				returned.add((Command<?, Key>) readCommand(index));
159 			} else {
160 				// If command does not exists, stop here, as next commands won't
161 				// be appliable
162 				break;
163 			}
164 		}
165 		return returned;
166 	}

213 	/**
214 	 * When trying to restore, we first try to load file given by
215 	 * {@link #getSpaceFile()}, create an object input stream on it, and read it
216 	 * content.
217 	 * 
218 	 * @param storageSpace
219 	 *            default storage space is none was memorized
220 	 * @return used storage space
221 	 */
222 	@Override
223 	public <Key extends Serializable> StorageSpace<Key> restore(
224 			StorageSpace<Key> storageSpace) {
225 		StorageSpace<Key> returned = storageSpace;
226 		File spaceFile = getSpaceFile();
227 		if (spaceFile.exists()) {
228 			ObjectInputStream stream = null;
229 			try {
230 				stream = new ObjectInputStream(
231 						new FileInputStream(spaceFile));
232 				// First object to read is space persister config
233 				config = (DefaultSpacePersisterConfig) stream.readObject();
234 				// Then we have to read full space data
235 				returned = (StorageSpace<Key>) stream
236 						.readObject();
237 			} catch (Exception e) {
238 				logger.log(Level.INFO, "space was not readable", e);
239 			} finally {
240 				try {
241 					stream.close();
242 				} catch (IOException e) {
243 					e.printStackTrace();
244 				}
245 			}
246 		}
247 		// Now, load and apply all remaining commands
248 		Collection<Command<?, Key>> toApply = loadCommands(
249 				config.getSpacePersistedAt(), Long.MAX_VALUE);
250 		config.setCommandIndex(config.getSpacePersistedAt()+toApply.size());
251 		for (Command<?, Key> command : toApply) {
252 			command.execute(returned);
253 		}
254 		// Finally, re-persist all to have a compressed and consistent space
255 		save(returned);
256 		return returned;
257 	}

259 	/**
260 	 * Save storage space and all required data
261 	 */
262 	public <Key extends Serializable> void save(StorageSpace<Key> storageSpace) {
263 		// Now remove all command logs
264 		for(long index=config.getSpacePersistedAt(); index<=config.getCommandIndex(); index++) {
265 			buildCommandLogFileName(index).delete();
266 			config.setSpacePersistedAt(index);
267 			updateSpaceFile(storageSpace);
268 		}
269 	}

271 	/**
272 	 * Update space file with new values of both config and storageSapce
273 	 * @param <Key>
274 	 * @param storageSpace
275 	 */
276 	private <Key extends Serializable> void updateSpaceFile(StorageSpace<Key> storageSpace) {
277 		File spaceFile = getSpaceFile();
278 		ObjectOutputStream persister = null;
279 		try {
280 			persister = new ObjectOutputStream(new FileOutputStream(spaceFile));
281 			persister.writeObject(config);
282 			persister.writeObject(storageSpace);
283 		} catch (IOException e) {
284 			e.printStackTrace();
285 		} finally {
286 			try {
287 				persister.close();
288 			} catch (IOException e) {
289 				e.printStackTrace();
290 			}
291 		}
292 	}
