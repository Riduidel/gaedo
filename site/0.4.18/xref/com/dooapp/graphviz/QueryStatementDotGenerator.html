116 	/**
117 	 * Adds a node id to the stack, and branch both its start and end view to given parent nodes.
118 	 * 
119 	 * @param nodeName node name to add
120 	 * @param startNode start node, may be null
121 	 * @param endNode end node may be null
122 	 */
123 	private void addNode(String nodeName, String startNode, String endNode) {
124 		String id = nodeName;
125 		if(startNode!=null)
126 			sOut.append("\t").append(startNode).append(DOT_ARROW).append(id).append(";\n");
127 		if(endNode!=null)
128 			sOut.append("\t").append(id+END_SUFFIX).append(DOT_ARROW).append(endNode).append(";\n");
129 		// Push node id
130 		nodes.push(id);
131 	}

138 	public void addQueryExpression() {
139 		// Stack only contains query statement, we branch on it, then on SortingExpression
140 		if(nodes.size()==1) {
141 			addNode(QUERY_EXPRESSION,QUERY_STATEMENT,SORTING_EXPRESSION);
142 		} else {
143 			addNode(QUERY_EXPRESSION+"_"+(nodeIndex++));
144 		}
145 		setQueryExpressionAttributes(nodes.peek(), false);
146 		setQueryExpressionAttributes(nodes.peek()+END_SUFFIX, false);
147 		setShape(nodes.peek(), "box");
148 		setShape(nodes.peek()+END_SUFFIX, "box");
149 	}

215 	 * Put a node atop of node stack without adding it to stack. it is useful for terminal expression nodes
216 	 * @param nodeName
217 	 * @param expectedParent TODO
218 	 * @return the node id
219 	 */
220 	private String putNode(String nodeName, String expectedParent) {
221 		String id = nodeName+"_"+(nodeIndex++);
222 		if(!nodes.peek().startsWith(expectedParent)) {
223 			addQueryExpression();
224 		}
225 		sOut.append("\t").append(nodes.peek()).append(DOT_ARROW).append(id).append(";\n");
226 		sOut.append("\t").append(id).append(DOT_ARROW).append(nodes.peek()+END_SUFFIX).append(";\n");
227 		return id;
