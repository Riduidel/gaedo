43  		CompilationUnit cu = new CompilationUnit();
44  		// set the package
45  		cu.setPackage(new PackageDeclaration(ASTHelper.createNameExpr(informerInfos.classPackage)));
46  
47  		// Finally add imports (they may have been "improved" by informers generation)
48  		List<ImportDeclaration> baseImports = informerInfos.imports;
49  		// Extracting effective imports
50  		Collection<String> imports = new LinkedList<String>();
51  		
52  		// Add current package to import for resolution (but do not forget to remove it before writing the effective imports)
53  		imports.add(informerInfos.classPackage);
54  		imports.add("java.lang");
55  		for (ImportDeclaration d : baseImports) {
56  			imports.add(d.getName().toString());
57  		}
58  
59  		// create the type declaration
60  		ClassOrInterfaceDeclaration type = new ClassOrInterfaceDeclaration(ModifierSet.PUBLIC, true, informerInfos.getInformerName());
61  		List<ClassOrInterfaceType> extended = new LinkedList<ClassOrInterfaceType>();
62  		extended.add(new ClassOrInterfaceType(Informer.class.getSimpleName() + "<" + informerInfos.className + ">"));
63  		extended.add(new ClassOrInterfaceType(informerInfos.getAbstractInformerName()));
64  		type.setExtends(extended);
65  		type.setJavaDoc(new JavadocComment("\n" + 
66  						"Informer for {@link " + informerInfos.className + "}\n" +
67  						"This is the public Informer class one must use to access dynamic properties methods." + 
68  						"@author InformerMojos\n"));
69  		List<AnnotationExpr> annotations = new LinkedList<AnnotationExpr>();
70  		// Constructing generated annotation value
71  		List<MemberValuePair> parameters = new LinkedList<MemberValuePair>();
72  		parameters.add(new MemberValuePair("date", new StringLiteralExpr(
73  						javax.xml.bind.DatatypeConverter.printDateTime(GregorianCalendar.getInstance()))));
74  		parameters.add(new MemberValuePair("comments", new StringLiteralExpr("generated by gaedo-informer-generator")));
75  		List<Expression> values = new LinkedList<Expression>();
76  		values.add(new StringLiteralExpr(informerInfos.getQualifiedClassName()));
77  		parameters.add(new MemberValuePair("value",

79  		NormalAnnotationExpr generated = new NormalAnnotationExpr(ASTHelper.createNameExpr(Generated.class.getName()), parameters);
80  		annotations.add(generated);
81  		type.setAnnotations(annotations);
82  		ASTHelper.addTypeDeclaration(cu, type);
83  		
84  		imports.remove(informerInfos.classPackage);
85  		imports.remove("java.lang");

87  		baseImports = new LinkedList<ImportDeclaration>();
88  		for(String name : imports) {
89  			baseImports.add(new ImportDeclaration(ASTHelper.createNameExpr(name), false, false));

92  		baseImports.add(new ImportDeclaration(ASTHelper.createNameExpr(ObjectFieldInformer.class.getPackage().getName()), false, true));

98  		CompilationUnit cu = new CompilationUnit();
99  		// set the package
100 		cu.setPackage(new PackageDeclaration(ASTHelper.createNameExpr(informerInfos.classPackage)));
101 
102 		// Finally add imports (they may have been "improved" by informers generation)
103 		List<ImportDeclaration> baseImports = informerInfos.imports;
104 		// Extracting effective imports
105 		Collection<String> imports = new LinkedList<String>();
106 		
107 		// superclasspackage will stay null as long as no import has been found for a class which name ends with ".superClassName"
108 		// Obvioulsy, if none is found, a warning is written in log
109 		String superClassPackage = null;
110 		
111 		// Add current package to import for resolution (but do not forget to remove it before writing the effective imports)
112 		imports.add(informerInfos.classPackage);
113 		imports.add("java.lang");
114 		String superClassSuffix = null;
115 		if(informerInfos.superClassName!=null)
116 			superClassSuffix = "."+informerInfos.superClassName;
117 		for (ImportDeclaration d : baseImports) {
118 			String importName = d.getName().toString();
119 			imports.add(importName);
120 			if(informerInfos.superClassName!=null) {
121 				if(importName.endsWith(superClassSuffix)) {
122 					if(superClassPackage!=null) {
123 						log.warn("more than one imports of "+informerInfos.getQualifiedClassName()+" contains superclass name. Last one found is "+importName+" This confuses gaedo-informer-generator");
124 					}
125 					superClassPackage = importName.substring(0, importName.lastIndexOf(superClassSuffix));
126 				}
127 			}
128 		}
129 
130 		// create the type declaration
131 		ClassOrInterfaceDeclaration type = new ClassOrInterfaceDeclaration(ModifierSet.PUBLIC, true, informerInfos.getAbstractInformerName());
132 		if(informerInfos.superClassName!=null && !"Object".equals(informerInfos.superClassName)) {
133 			List<ClassOrInterfaceType> extended = new LinkedList<ClassOrInterfaceType>();
134 			String superInformer = InformerInfos.buildAbstractInformerName(informerInfos.superClassName);
135 			// do not forget to add import for that class (see https://github.com/Riduidel/gaedo/issues/20)
136 			if(superClassPackage!=null) {
137 				imports.add(superClassPackage+"."+superInformer);
138 			}
139 			extended.add(new ClassOrInterfaceType(superInformer));
140 			type.setExtends(extended);
141 		}
142 		type.setJavaDoc(new JavadocComment("\n" + 
143 						"Informer method container for {@link " + informerInfos.className + "}\n" +
144 						"This interface is to be used only by gaedo code. Its only role is to provide consistent method hierarchy.\n" +
145 						"As a consequence, refering to it directly has not the slightest interest and should never been done in user code." + 
146 						"@author InformerMojos\n"));
147 		List<AnnotationExpr> annotations = new LinkedList<AnnotationExpr>();
148 		// Constructing generated annotation value
149 		List<MemberValuePair> parameters = new LinkedList<MemberValuePair>();
150 		parameters.add(new MemberValuePair("date", new StringLiteralExpr(
151 						javax.xml.bind.DatatypeConverter.printDateTime(GregorianCalendar.getInstance()))));
152 		parameters.add(new MemberValuePair("comments", new StringLiteralExpr("generated by gaedo-informer-generator")));
153 		List<Expression> values = new LinkedList<Expression>();
154 		values.add(new StringLiteralExpr(informerInfos.getQualifiedClassName()));
155 		parameters.add(new MemberValuePair("value", 
156 						new ArrayInitializerExpr(values)));
157 		NormalAnnotationExpr generated = new NormalAnnotationExpr(ASTHelper.createNameExpr(Generated.class.getName()), parameters);
158 		annotations.add(generated);
159 		type.setAnnotations(annotations);
160 		ASTHelper.addTypeDeclaration(cu, type);
161 		for (PropertyInfos infos : informerInfos.properties) {
162 			// create a method
163 			MethodDeclaration method = new MethodDeclaration(ModifierSet.PUBLIC, ASTHelper.VOID_TYPE, "get" + Utils.uppercaseFirst(infos.name));
164 			String informerTypeFor = InformerTypeFinder.getInformerTypeFor(resolvedInformers, qualifiedEnums, imports, infos);
165 			method.setType(new ClassOrInterfaceType(informerTypeFor));
166 			method.setJavaDoc(new JavadocComment(infos.generateJavadoc(informerInfos, informerTypeFor)));
167 			ASTHelper.addMember(type, method);
168 		}
169 		
170 		imports.remove(informerInfos.classPackage);
171 		imports.remove("java.lang");
172 
173 		baseImports = new LinkedList<ImportDeclaration>();
174 		for(String name : imports) {
175 			baseImports.add(new ImportDeclaration(ASTHelper.createNameExpr(name), false, false));
176 		}
177 		baseImports.add(new ImportDeclaration(ASTHelper.createNameExpr(Informer.class.getCanonicalName()), false, false));
178 		baseImports.add(new ImportDeclaration(ASTHelper.createNameExpr(ObjectFieldInformer.class.getPackage().getName()), false, true));
179 		cu.setImports(baseImports);
180 		return cu;
181 	}
