18  /**
19   * Entity filler will take field from object and put them in the entity in the
20   * most correct way. Notice that, since this class provides direct access to Google Datastore, it is not accessible to external code.
21   * 
22   * @author Nicolas
23   * 
24   */
25  class EntityFiller<DataType, InformerType extends Informer<DataType>>
26  		implements EntityObjectMapper<DataType> {

45  	/**
46  	 * When mapping, some particular cases are handled. If value type is a
47  	 * collection type, some specific magick is done. If value type is managed
48  	 * by a service, we ensure value for object is already persisted, elsewhere
49  	 * we persist it. Then we ask its key to the appropriate service and store
50  	 * it.
51  	 */
52  	@Override
53  	public void map(Entity entity, DataType object, String key, Property value) {
54  		try {
55  			Type valueType = value.getGenericType();
56  			Object valueForObject = null;
57  			valueForObject = value.get(object);
58  			fillEntity(entity, key, valueType, valueForObject);
59  		} catch (Exception e) {
60  			throw new UnableToGetFieldException(e, value);
61  		}
62  	}

64  	/**
65  	 * Fill entity with infos from given property
66  	 * 
67  	 * @param entity
68  	 *            output entity
69  	 * @param key
70  	 *            save key
71  	 * @param value
72  	 *            value field
73  	 * @param valueForObject
74  	 *            value for the given object
75  	 * @throws IllegalAccessException
76  	 */
77  	private void fillEntity(Entity entity, String key, Type valueType,
78  			Object valueForObject) throws IllegalAccessException {
79  		if (valueType instanceof Class) {
80  			Class cls = (Class) valueType;
81  			if(cls.isEnum()) {
82  				// For an enum, we only need its name (obfuscation won't be supported here, so long proGuard !)
83  				entity.setProperty(key, ((Enum) valueForObject).name());
84  			} else {
85  				if (repository
86  						.containsKey((Class<?>) valueType)
87  						&& valueForObject != null) {
88  					valueForObject = getValueFromDomain((Class<?>) valueType,
89  							valueForObject);
90  				}
91  				entity.setProperty(key, valueForObject);
92  			}
93  		} else if (valueType instanceof ParameterizedType) {
94  			ParameterizedType parameterizedType = (ParameterizedType) valueType;
95  			if (Collection.class.isAssignableFrom((Class<?>) parameterizedType
96  					.getRawType())) {
97  				// A collection is genericized using only one type argument
98  				setCollectionProperties(entity, key, parameterizedType
99  						.getActualTypeArguments()[0],
100 						(Collection<?>) valueForObject);
101 			} else if (Map.class.isAssignableFrom((Class<?>) parameterizedType
102 					.getRawType())) {
103 				// A collection is genericized using only one type argument
104 				setMapProperties(entity, key, parameterizedType
105 						.getActualTypeArguments()[0], parameterizedType
106 						.getActualTypeArguments()[1],
107 						(Map<?, ?>) valueForObject);
108 			}
109 
110 		}
111 	}

113 	/**
114 	 * Create object sub-entities for each collection element. The main issue of
115 	 * this method is to not fall into the hell of re-writing all elements each
116 	 * time, while not forgeting to write anyone.
117 	 * 
118 	 * @param entity
119 	 *            output entity
120 	 * @param key
121 	 *            entity base key
122 	 * @param value
123 	 *            field representing the collection
124 	 * @param valueForObject
125 	 *            value for this object (in this case, it is a
126 	 *            {@link Collection})
127 	 * @throws IllegalAccessException
128 	 */
129 	private void setCollectionProperties(Entity entity, String key,
130 			Type valueType, Collection<?> valueForObject)
131 			throws IllegalAccessException {
132 		Key root = entity.getKey();
133 		if(valueForObject==null) {
134 			fillEntity(entity, key + Utils.SIZE, Integer.class, 0);
135 		} else {
136 			fillEntity(entity, key + Utils.SIZE, Integer.class, valueForObject
137 					.size());
138 			if(((valueForObject instanceof LazyLoadable) && ((LazyLoadable) valueForObject).isLoaded()) || !(valueForObject instanceof LazyLoadable)) {
139 				int index = 0;
140 				for (Object element : valueForObject) {
141 					// Keys use 1-starting indexes !
142 					Key current = KeyFactory.createKey(root, key, index + 1);
143 					Entity nodeEntity = new Entity(current);
144 					fillEntity(nodeEntity, Utils.COLLECTION_VALUE_PROPERTY, valueType, element);
145 					datastore.put(nodeEntity);
146 					index++;
147 				}
148 			}
149 		}
150 	}

152 	private void setMapProperties(Entity entity, String key, Type keyType,
153 			Type valueType, Map<?, ?> valueForObject)
154 			throws IllegalAccessException {
155 		Key root = entity.getKey();
156 		if(valueForObject==null) {
157 			fillEntity(entity, key + Utils.SIZE, Integer.class, 0);
158 		} else {
159 			fillEntity(entity, key + Utils.SIZE, Integer.class, valueForObject
160 					.size());
161 			if(((valueForObject instanceof LazyLoadable) && ((LazyLoadable) valueForObject).isLoaded()) || !(valueForObject instanceof LazyLoadable)) {
162 				int index = 0;
163 				for (Map.Entry<?, ?> element : valueForObject.entrySet()) {
164 					// Keys use 1-starting indexes !
165 					Key current = KeyFactory.createKey(root, key, index + 1);
166 					Entity nodeEntity = new Entity(current);
167 					fillEntity(nodeEntity, Utils.MAP_KEY_PROPERTY, keyType, element.getKey());
168 					fillEntity(nodeEntity, Utils.MAP_VALUE_PROPERTY, valueType, element.getValue());
169 					datastore.put(nodeEntity);
170 					index++;
171 				}
172 			}
173 		}
174 	}

176 	/**
177 	 * Here, the DataType may be confusin (it even confuses me) since we call a
178 	 * method from another service by using the same DataType. However, never
179 	 * forget that a cast is always removed after compilation, it's just an
180 	 * artifice.
181 	 * 
182 	 * @param valueType
183 	 * @param valueForObject
184 	 * @return
185 	 * @throws IllegalAccessException
186 	 */
187 	/*
188 	 * For a strange reason compiling this with maven fails with a mysterious
189 	 * message: inconvertible types found
190 	 * 
191 	 * Fast and ugly solution was obviously to get rid of generics
192 	 */
193 	@SuppressWarnings("unchecked")
194 	private Object getValueFromDomain(Class<?> valueType, Object valueForObject)
195 			throws IllegalAccessException {
196 		DatastoreFinderService valueService = (DatastoreFinderService) repository
197 				.get(valueType);
198 		DataType typedValue = (DataType) valueForObject;
199 		if(!(valueService.getIdManager().hasKey(valueService.getKind(), typedValue))) {
200 			typedValue = (DataType) valueService.create(typedValue);
201 		}
202 		valueForObject = valueService.getIdManager().getKey(valueService.getKind(), typedValue);
203 		return valueForObject;
204 	}
