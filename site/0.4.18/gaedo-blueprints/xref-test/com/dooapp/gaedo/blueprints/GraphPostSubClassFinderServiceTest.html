176 		Vertex postVertex = postSubService.getVertexFor(newOne, CascadeType.REFRESH, new TreeMap<String, Object>());

179 			((TransactionalGraph) environment.getGraph()).startTransaction();
180 		Vertex valueVertex = environment.getGraph().addVertex("new-text-vertex");
181 		valueVertex.setProperty(Properties.kind.name(), Kind.uri.name());
182 		String BUG_31_URI = "https://github.com/Riduidel/gaedo/issues/31";
183 		valueVertex.setProperty(Properties.value.name(), BUG_31_URI);
184 		
185 		// Have you notice this property is annotated ?
186 		String textPropertyName = "post:text";
187 		Iterable<Edge> previous = postVertex.getOutEdges(textPropertyName);
188 		for(Edge e : previous) {
189 			environment.getGraph().removeEdge(e);
190 		}
191 		
192 		Edge edgeToURI = environment.getGraph().addEdge(textPropertyName, postVertex, valueVertex, textPropertyName);
193 		String predicateProperty = GraphUtils.asSailProperty(textPropertyName);
194 		edgeToURI.setProperty(GraphSail.PREDICATE_PROP, predicateProperty);
195 		edgeToURI.setProperty(GraphSail.CONTEXT_PROP, "U "+GraphUtils.GAEDO_CONTEXT);
196 		edgeToURI.setProperty(GraphSail.CONTEXT_PROP + GraphSail.PREDICATE_PROP, 
197 						edgeToURI.getProperty(GraphSail.CONTEXT_PROP).toString()+" "+
198 						edgeToURI.getProperty(GraphSail.PREDICATE_PROP).toString());
199 
200 		
201 		Post loaded = postSubService.findById(newOne.id);
202 		
203 		assertThat(loaded.text, Is.is(BUG_31_URI));
