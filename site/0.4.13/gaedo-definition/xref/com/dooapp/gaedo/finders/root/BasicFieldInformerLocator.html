54  	 * Get informer for given type, by reflective upon its attribute
55  	 * @param property source property
56  	 * @param genericType generic type of property
57  	 * @return
58  	 */
59  	protected FieldInformer getInformerFor(Property property, Type genericType) {
60  		if (genericType instanceof Class) {
61  			// Any other case is a class one (look at
62  			// http://java.sun.com/j2se/1.5.0/docs/api/java/lang/reflect/Type.html
63  			// for more info)
64  			Class<?> clazz = (Class<?>) genericType;
65  			Class<? extends FieldInformer> informerClass = getInformerForClass(clazz);
66  			// This class must have a constructor using one Property parameter, elsewhere mayhem may come
67  			Constructor<? extends FieldInformer> used;
68  			try {
69  				used = informerClass.getDeclaredConstructor(Property.class);
70  				return used.newInstance(property);
71  			} catch (Exception e) {
72  				throw new UnableToCreateFieldInformerException(clazz, informerClass, property.getClass(), e);
73  			}
74  		} else if(genericType instanceof ParameterizedType) {
75  			// Too lazy to write it all
76  			ParameterizedType pType = (ParameterizedType) genericType;
77  			return getInformerFor(property, pType.getRawType());
78  		}
79  		return null;
80  	}

83  	 * Helper method allowing somee overriding of class localization
84  	 * @param clazz
85  	 * @return
86  	 */
87  	protected Class<? extends FieldInformer> getInformerForClass(Class<?> clazz) {
88  		Class<? extends FieldInformer> existingOne = hierarchicalySearchForInformer(clazz);
89  		if(existingOne==null)
90  			throw new NoInformerForThisTypeException(clazz);
91  		else
92  			return existingOne;
93  	}

95  	/**
96  	 * Search in all superclasses of given class for informers
97  	 * @param clazz source class
98  	 * @return informer, if found, null elsewhen.
99  	 */
100 	private Class<? extends FieldInformer> hierarchicalySearchForInformer(Class<?> clazz) {
101 		Class<? extends FieldInformer> existingOne = null;
102 		if(existingInformerMappings.containsKey(clazz)) {
103 			return existingInformerMappings.get(clazz);
104 		} else {
105 			Map<Class<?>, Class<? extends FieldInformer>> foundInformers = new HashMap<Class<?>, Class<? extends FieldInformer>>();
106 			for(Class<?> interfazz : clazz.getInterfaces()) {
107 				if(existingInformerMappings.containsKey(clazz))
108 					foundInformers.put(interfazz, existingInformerMappings.get(clazz));
109 			}
110 			if(foundInformers.size()==1) {
111 				return foundInformers.entrySet().iterator().next().getValue();
112 			} else if(foundInformers.size()>1) {
113 				throw new ConflictingInformersForThisTypeException(clazz, foundInformers);
114 			} else if(Object.class.equals(clazz)) {
115 				return null;
116 			} else {
117 				if(clazz.getSuperclass()!=null) {
118 					return hierarchicalySearchForInformer(clazz.getSuperclass());
119 				} else {
120 					return null;
121 				}
122 			}
123 		}
124 	}
