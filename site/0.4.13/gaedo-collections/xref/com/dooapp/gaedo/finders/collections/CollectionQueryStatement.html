82  	/**

86  	private List<DataType> selectAll() {
87  		List<DataType> returned = new ArrayList<DataType>();
88  		Matcher<DataType> expression = createMatcher();
89  		for (DataType element : data) {
90  			if (expression.matches(element)) {
91  				returned.add(element);
92  			}
93  		}

104 		Matcher<DataType> matcher = new Matcher<DataType>();
105 		// When the expression accepts the matcher, the matcher will construct
