30  		PropertyChangeEmitter {
31  	public static class NoSuchDataException extends CrudServiceException {
32  
33  		private final Object unknownData;
34  
35  		public Object getUnknownData() {
36  			return unknownData;
37  		}
38  
39  		public NoSuchDataException(Object toUpdate) {
40  			super("data " + toUpdate.toString()
41  					+ " does not exists in this service");
42  			this.unknownData = toUpdate;
43  		}
44  
45  	}
46  
47  	private final Collection<DataType> data = new LinkedList<DataType>();
48  
49  	private final PropertyChangeEmitter support = new PropertyChangeEmitterImpl();
50  
51  	/**
52  	 * Build a collection backed finder
53  	 * 
54  	 * @param containedClass
55  	 * @param informerClass
56  	 * @see AbstractFinderService
57  	 */
58  	public CollectionBackedFinderService(
59  			Class<DataType> containedClass, Class<InformerType> informerClass,
60  			InformerFactory factory) {
61  		super(containedClass, informerClass, factory);
62  	}
63  
64  	public DataType create(DataType toCreate) {
65  		data.add(toCreate);
66  		return toCreate;
67  	}
68  
69  	public void delete(DataType toDelete) {
70  		data.remove(toDelete);
71  	}
72  
73  	public DataType update(DataType toUpdate) {
74  		if (!data.contains(toUpdate)) {
75  			throw new NoSuchDataException(toUpdate);
76  		}
77  		return toUpdate;
78  	}
79  
80  	/**
81  	 * Creates the query statement used a copy of current data. As a
82  	 * consequence, long calls may result in inaccurate data set.
83  	 */
84  	@Override
85  	protected QueryStatement<DataType, InformerType> createQueryStatement(
86  			QueryBuilder<InformerType> query) {
87  		return new CollectionQueryStatement<DataType, InformerType>(query,
88  				getInformer(), new LinkedList<DataType>(data), support);
89  	}
90  
91  	/**

93  	 *         upon {@link #data}
94  	 */
95  	public Iterable<DataType> findAll() {

97  	}
