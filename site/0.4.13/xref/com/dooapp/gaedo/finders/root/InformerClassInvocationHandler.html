32  	static final String SYNTHETIC_GETTER_PREFIX = "get";
33  	/**
34  	 * Utility method obtaining the field informer associated to a particular method name.
35  	 * @param <ContainedType>
36  	 * @param method method
37  	 * @param methodName
38  	 * @param realInformer
39  	 * @return
40  	 */
41  	public static <ContainedType> FieldInformer methodNameToFieldInformer(Method method,
42  			Informer<ContainedType> realInformer) {
43  		String methodName = method.getName();
44  		if(methodName.startsWith(SYNTHETIC_GETTER_PREFIX)) {
45  			String fieldName = methodName.substring(SYNTHETIC_GETTER_PREFIX
46  					.length());
47  			if (fieldName.length() > 0) {
48  				String realFieldName = fieldName.substring(0, 1).toLowerCase();
49  				if (fieldName.length() > 1)
50  					realFieldName += fieldName.substring(1);
51  				try {
52  					return realInformer.get(realFieldName);
53  				} catch(NoSuchFieldInHierarchyException e) {
54  					throw new MethodConstructedOnMissingField(method, realFieldName, e);
55  				}
56  			} else {
57  				throw new EmptyGetDontWorkException(method);
58  			}
59  		} else {
60  			throw new BadPrefixException(method);
61  		}
62  	}

126 		Class<?> declaringClass = method.getDeclaringClass();
127 		if(declaringClass.equals(FieldProjector.class)) {

136 		} else if (declaringClass.equals(FieldInformerAPI.class)) {
137 			return new CallMethodResolver(method) {

140 					return invokeAPIMethod(method, invokedArgs);

142 			};
143 		} else if (declaringClass.equals(Validable.class)) {
144 			return new CallMethodResolver(method) {

147 					return invokeValidableMethod(method, invokedArgs);
148 				}
149 			};
150 		} else if (declaringClass.equals(Proxied.class)) {
151 			return new CallMethodResolver(method) {
152 				@Override

154 					return invokeProxiedMethod(method, invokedArgs);
155 				}
156 			};
157 		} else if (declaringClass.isAssignableFrom(Informer.class)) {
158 			return new CallMethodResolver(method) {

161 					return invokeInformerMethod(method, invokedArgs);

163 			};
164 		} else  {

166 		}

172 	 * Invokes a method on this object, which IS validable
173 	 * @param invokedMethod
174 	 * @param invokedArgs
175 	 * @return
176 	 * @category invoke
177 	 */
178 	protected Object invokeValidableMethod(Method invokedMethod, Object[] invokedArgs) {
179 		try {
180 			return invokedMethod.invoke(this, invokedArgs);
181 		} catch (IllegalArgumentException e) {
182 			throw new InvocationFailedException(e);
183 		} catch (IllegalAccessException e) {
184 			throw new InvocationFailedException(e);
185 		} catch (InvocationTargetException e) {
186 			if(e.getTargetException() instanceof CrudServiceException)
187 				throw (CrudServiceException) e.getTargetException();
188 			throw new InvocationFailedException(e);
189 		}

193 	 * Create method for sub-informer class declarations. Those are the "magical" ones which bind field names in method name to
194 	 * Informers.
195 	 * @see #methodNameToFieldInformer(Method, Informer)
196 	 * @param method input method
197 	 * @return a method resolving call to an informer
198 	 */
199 	private MethodResolver createSubInformerResolvers(Method method) {
200 		// First thing to check is method startup
201 		FieldInformer returned = methodNameToFieldInformer(method, realInformer);
202 		Class<?> returnType = method.getReturnType();
203 		if (!returnType.isAssignableFrom(returned.getClass())) {
204 			throw new ReturnTypeMismatchException(method,
205 					returned.getClass(), returnType);
206 		}
207 		return new SubInformerMethodResolver(returned, realInformer);
208 	}
