133 		if(generateMappings) {

140 				}

142 		}
143 		getLog().info("examining all classes looking for the beans ones");
144 		Map<String, Class> resolvedInformers = com.dooapp.gaedo.informer.Utils.createResolvedInformers();
145 		Collection<File> sources = getSourceFiles(project.getCompileSourceRoots());
146 		Collection<String> qualifiedEnums = findQualifiedEnums(sources);
147 		qualifiedEnums.addAll(Arrays.asList(enums));
148 		getLog().info("there seems to be those enums in the path : "+qualifiedEnums);
149 		Collection<CompilationUnit> matching = findMatchingSources(sources);
150 		// now generate informer for each interesting file (but for that we have
151 		// to know if file is interesting
152 		Collection<String> excludedProperties = new LinkedList<String>();
153 		excludedProperties.addAll(Arrays.asList(propertiesExcludes));
154 		try {
155 			for (CompilationUnit u : matching) {
156 				buildInformerFor(u, qualifiedEnums, resolvedInformers, excludedProperties);
157 			}
158 			// Finally add generated sources to sources for project
159 			project.getCompileSourceRoots().add(output.getCanonicalPath());
160 		} catch (IOException e) {
161 			throw new MojoExecutionException("unable to resolve output folder to a canonical path", e);
162 		}
163 		if(generateMappings) {
164 			informerMappings.getParentFile().mkdirs();
165 			try {
166 				currentInformerMappings.store(new FileOutputStream(informerMappings), "informer mappings generated by Gaedo Informer generator mojo");
167 				getLog().info("written informer mappings in "+informerMappings.getAbsolutePath());
168 			} catch (Exception e) {

174 	public Collection<String> findQualifiedEnums(Collection<File> sources) {
175 		Collection<String> returned = new LinkedList<String>();
176 		if(enums!=null) {
177 			returned.addAll(Arrays.asList(enums));
178 		}
179 		for (File f : sources) {
180 			try {
181 				CompilationUnit searched = JavaParser.parse(f);
182 				EnumScanner enums = new EnumScanner();
183 				enums.visit(searched, null);
184 				if (enums.getQualifiedName()!=null) {
185 					returned.add(enums.getQualifiedName());
186 				}
187 			} catch (Exception e) {
188 				getLog().error("file " + f + " couldn't be parsed", e);
189 			}
190 		}
191 		return returned;
192 	}

215 		getLog().info("generating informer for "+informerInfos.classPackage+"."+informerInfos.className);
216 		getLog().debug("it should have as FieldInformers "+informerInfos.properties);
217 		CompilationUnit cu = InformerTextGenerator.generateVisibleInformer(informerInfos, qualifiedEnums, resolvedInformers);
218 		File effective = new File(baseOutput+"/"+informerInfos.classPackage.replace('.', '/')+"/"+informerInfos.getInformerName()+".java");
219 		effective.getParentFile().mkdirs();

221 		cu = InformerTextGenerator.generateAbstractInformer(informerInfos, qualifiedEnums, resolvedInformers, getLog());
222 		effective = new File(baseOutput+"/"+informerInfos.classPackage.replace('.', '/')+"/"+informerInfos.getAbstractInformerName()+".java");

243 	 * Extract from the whole collection of files representing java sources the
244 	 * smaller collection of copmilation units representing classes matching
245 	 * with either {@link #requiredAnnotations} or {@link #requiredInterfaces}
246 	 * 
247 	 * @param sources
248 	 * @return
249 	 */
250 	public List<CompilationUnit> findMatchingSources(Collection<File> sources) {
251 		List<CompilationUnit> returned = new LinkedList<CompilationUnit>();
252 		for (File f : sources) {
253 			try {
254 				CompilationUnit searched = JavaParser.parse(f);
255 				ScanningMatcher interfaces = new RequiredInterfacesScanner(requiredInterfaces);
256 				ScanningMatcher annotations = new RequiredAnnotationsScanner(requiredAnnotations);
257 				if (interfaces.matches(searched) && annotations.matches(searched)) {
258 					returned.add(searched);
259 				}
260 			} catch (Exception e) {
261 				getLog().error("file " + f + " couldn't be parsed", e);
262 			}
263 		}
264 		return returned;
265 	}

284 	/**
285 	 * Scans a single directory.
286 	 * 
287 	 * @param root
288 	 *            Directory to scan
289 	 * @param writer
290 	 *            Where to write the source list
291 	 * @throws MojoExecutionException
292 	 *             in case of IO errors
293 	 * @return collection of files matching criterias
294 	 */
295 	private Collection<File> scan(File root) throws MojoExecutionException {
296 		Collection<File> returned = new LinkedList<File>();
297 
298 		if (!root.exists()) {
299 			getLog().warn(root + " is not an existing directory");
300 			return returned;
301 		}
302 
303 		getLog().info("scanning source file directory '" + root + "'");
304 
305 		final DirectoryScanner directoryScanner = new DirectoryScanner();
306 		directoryScanner.setIncludes(includes);
307 		directoryScanner.setExcludes(excludes);
308 		directoryScanner.setBasedir(root);
309 		directoryScanner.scan();
310 
311 		for (String fileName : directoryScanner.getIncludedFiles()) {
312 			File file = new File(root, fileName);
313 			returned.add(file);
314 		}
315 		return returned;
316 	}
