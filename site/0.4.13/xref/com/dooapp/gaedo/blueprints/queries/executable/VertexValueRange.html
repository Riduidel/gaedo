39  	private void getValues(Entry<Iterable<Vertex>, Iterable<Property>> entry, Iterator<Property> pathIterator, List<Vertex> vertexAccumulator) {
40  		if(pathIterator.hasNext()) {
41  			Property current = pathIterator.next();
42  			getValues(entry, pathIterator, vertexAccumulator);
43  			List<Vertex> toScan = new LinkedList<Vertex>(vertexAccumulator);
44  			// clear parameter to let it be populated by result
45  			vertexAccumulator.clear();
46  			for(Vertex v : toScan) {
47  				Iterable<Edge> edges = v.getInEdges(GraphUtils.getEdgeNameFor(current));
48  				for(Edge e : edges) {
49  					vertexAccumulator.add(e.getOutVertex());
50  				}
51  			}
52  		} else {
53  			for(Vertex v : entry.getKey()) {
54  				vertexAccumulator.add(v);
55  			}
56  		}

69  	/**
70  	 * Recursive method computing, using all the possible paths, the edge count linking 
71  	 * @param entry evaluated entry
72  	 * @param edgesCount current value of edges count, an atomic long is used to not use primitive type (which is immutable) while guaranteeing that value will be safely changed.
73  	 * @param verticesToCountEdgesIn 
74  	 * @param pathIterator that iterator browses the path and at each path element calls this very method to recurse over path
75  	 * @param usingProperty property that should be used to find the vertex in the graph (instead of simply relying upon valye type (which is especially useful for numbers). This property can be null
76  	 * @return
77  	 */
78  	private VertexValueRange findBestMatch(Entry<Iterable<Vertex>, Iterable<Property>> entry, AtomicLong edgesCount, ArrayList<Vertex> verticesToCountEdgesIn, Iterator<Property> pathIterator) {
79  		if(pathIterator.hasNext()) {
80  			Property current = pathIterator.next();
81  			VertexValueRange matching = findBestMatch(entry, edgesCount, verticesToCountEdgesIn, pathIterator);
82  			// If previous level already returned this entryscore, there is no chance for new one to have less edges ... or we guess so
83  			if(matching==this) {
84  				return matching;
85  			} else {
86  				long totalEdgeCount = 0;
87  				List<Vertex> toScan = new LinkedList<Vertex>(verticesToCountEdgesIn);
88  				// clear parameter to let it be populated by result
89  				verticesToCountEdgesIn.clear();
90  				for(Vertex v : toScan) {
91  					Iterable<Edge> edges = v.getInEdges(GraphUtils.getEdgeNameFor(current));
92  					for(Edge e : edges) {
93  						verticesToCountEdgesIn.add(e.getOutVertex());
94  						totalEdgeCount = edgesCount.addAndGet(1);
95  						if(totalEdgeCount>this.edgesCount)
96  							return this;
97  					}
98  				}
99  				totalEdgeCount = edgesCount.get();
100 				if(totalEdgeCount<this.edgesCount) {
101 					// Due to previous if, we can guarantee matching is not this object, but the new one (for which we need to update edgesCount)
102 					return matching.withEdgesCount(totalEdgeCount);
103 				} else {
104 					return this;
105 				}
106 			}
107 		} else {
108 			// the vertices to scan consist only in entry key
109 			for(Vertex v : entry.getKey()) {
110 				verticesToCountEdgesIn.add(v);
111 			}
112 			edgesCount.set(verticesToCountEdgesIn.size());
113 			// End of the path, no ? then return supposed best match (which is, up to now, a new EntryScore)
114 			return new VertexValueRange().withEntry(entry).withEdgesCount(edgesCount.get());
115 		}
116 	}
