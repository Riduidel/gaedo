34  	public MapEntryTransformer() {

36  		try {
37  			Class<Map.Entry> interfaceClass = Map.Entry.class; 
38  			Class<WriteableKeyEntry> implementationClass = WriteableKeyEntry.class;
39  			PropertyDescriptor keyProperty = new PropertyDescriptor("key", interfaceClass.getDeclaredMethod("getKey"), implementationClass.getDeclaredMethod("setKey", Object.class));
40  			PropertyDescriptor valueProperty = new PropertyDescriptor("value", interfaceClass.getDeclaredMethod("getValue"), interfaceClass.getDeclaredMethod("setValue", Object.class));
41  			
42  			KEY_PROPERTY = new DescribedProperty(keyProperty, Map.Entry.class);
43  			VALUE_PROPERTY = new DescribedProperty(valueProperty, Map.Entry.class); 
44  			
45  			containedProperties.put(
46  							KEY_PROPERTY, 
47  							cascade);
48  			containedProperties.put(
49  							VALUE_PROPERTY, 
50  							cascade);
51  		} catch(Exception e) {
52  			throw new UnableToGetKeyOrValueProperty(e);
53  		}
54  	}
