67  	/**
68  	 * Get the contained property
69  	 * 
70  	 * @param entity
71  	 *            source entity
72  	 * @param entityName
73  	 *            entity property name (in the entity considered as a hash)
74  	 * @param valueType
75  	 *            value type
76  	 * @param defaultvalue
77  	 *            default value, given as a hint (may be null)
78  	 * @return
79  	 */
80  	private Object getContainedProperty(Entity entity, String entityName,
81  			Type valueType, Object defaultvalue) {
82  		Object property = entity.getProperty(entityName);
83  		if (valueType instanceof Class<?>) {
84  			Object castedProperty = null;
85  			Class<?> valueClass = (Class<?>) valueType;
86  			if(valueClass.isEnum()) {
87  				// Stored value is a string
88  				castedProperty = Enum.valueOf((Class<? extends Enum>)valueClass, property.toString());
89  			} else {
90  				if (repository.containsKey(valueClass)
91  						&& property != null) {
92  					property = getValueFromDomain(valueClass, property);
93  				}
94  				castedProperty = Utils.getCastedProperty(valueClass,
95  						property);
96  			}
97  			return castedProperty;
98  		} else if (valueType instanceof ParameterizedType) {
99  			ParameterizedType parameterizedType = (ParameterizedType) valueType;
100 			Type rawContainerType = parameterizedType.getRawType();
101 			if (rawContainerType instanceof Class) {
102 				Class<?> rawContainerClass = (Class<?>) rawContainerType;
103 				if (Collection.class.isAssignableFrom(rawContainerClass)) {
104 					return setCollectionProperties(entity, entityName,
105 							parameterizedType.getActualTypeArguments()[0],
106 							com.dooapp.gaedo.utils.Utils.generateCollection(rawContainerClass,
107 									(Collection<?>) property),
108 							rawContainerClass);
109 				} else if (Map.class.isAssignableFrom(rawContainerClass)) {
110 					return setMapProperties(
111 							entity,
112 							entityName,
113 							parameterizedType.getActualTypeArguments()[0],
114 							parameterizedType.getActualTypeArguments()[1],
115 							com.dooapp.gaedo.utils.Utils.generateMap(rawContainerClass, (Map<?, ?>) property),
116 							rawContainerClass);
117 				}
118 			}
119 		}
120 		throw new UnsupportedOperationException("the case of fields like "
121 				+ valueType.toString() + " has not yet been handled !");
122 	}

125 	 * Load collection from entity in target object. This class loads collection
126 	 * by intercepting them with a lazy loading strategy.
127 	 * 
128 	 * @param entity
129 	 *            source entity
130 	 * @param fieldName
131 	 * @param rawContainerClass
132 	 */
133 	private Collection<?> setCollectionProperties(Entity entity,
134 			String fieldName, Type valueType, Collection toReturn,
135 			Class<?> rawContainerClass) {
136 		Key root = entity.getKey();
137 		int collectionSize = (Integer) Utils.getCastedProperty(Integer.TYPE,
138 				entity.getProperty(fieldName + Utils.SIZE));
139 		// since elements are 1-indexed in datastore, we have to use this
140 		// special loop construct
141 		Collection<Key> toGrab = new LinkedList<Key>();
142 		for (int index = 0; index < collectionSize; index++) {
143 			toGrab.add(KeyFactory.createKey(root, fieldName, index + 1));
144 		}
145 		return (Collection<?>) Proxy.newProxyInstance(getClass()
146 				.getClassLoader(), new Class<?>[] { rawContainerClass, LazyLoadable.class },
147 				new CollectionLazyLoader(valueType, toReturn, toGrab));
148 	}

167 	private class MapLazyLoader implements InvocationHandler, LazyLoadable {
168 		private boolean loaded = false;
169 
170 		private Type keyType;
171 		private Type valueType;
172 		private Map toReturn;
173 		private Collection<Key> keys;
174 
175 		public MapLazyLoader(Type keyType, Type valueType, Map<?, ?> toReturn,
176 				Collection<Key> toGrab) {
177 			this.keyType = keyType;
178 			this.valueType = valueType;
179 			this.toReturn = toReturn;
180 			this.keys = toGrab;
181 		}
182 
183 		private void load() {
184 			if(!loaded) {
185 				Map<Key, Entity> values = datastore
186 						.get(keys);
187 				for (Key source : keys) {
188 					Entity nodeEntity = values.get(source);
189 					Object containedKey = getContainedProperty(nodeEntity, Utils.MAP_KEY_PROPERTY,
190 							keyType, null);
191 					Object containedValue = getContainedProperty(nodeEntity,
192 							Utils.MAP_VALUE_PROPERTY, valueType, null);
193 					toReturn.put(containedKey, containedValue);
194 				}
195 				loaded = true;
196 			}
197 		}
198 
199 		@Override
200 		public Object invoke(Object proxy, Method method, Object[] args)
201 				throws Throwable {
202 			if(method.getDeclaringClass().equals(LazyLoadable.class)) {
203 				return method.invoke(this, args);
204 			}
205 			// Added to fix save inconsistency (see issue #33)
206 			if ("size".equals(method.getName()) && !loaded) {
207 				return keys.size();
208 			}
209 			// It's time to really load source collection
210 			if (toReturn.size() == 0)
211 				load();
212 			return method.invoke(toReturn, args);
213 		}
214 
215 		@Override
216 		public boolean isLoaded() {
217 			return loaded;
218 		}
