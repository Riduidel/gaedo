14  /**
15   * This property change listener records some time statistics for queries being
16   * executed on datastore.
17   * 
18   * @author ndx
19   * 
20   */
21  public class QueryQuotaMonitorer implements PropertyChangeListener {
22  	private static final Logger logger = Logger.getLogger("quotas");
23  	private class QueryInfos {
24  		private String queryId;
25  		private long start;
26  
27  		public QueryInfos(String queryId) {
28  			this.queryId  = queryId;
29  			this.start = quotaService.getCpuTimeInMegaCycles();
30  		}
31  
32  		public String terminate() {
33  			long end = quotaService.getCpuTimeInMegaCycles();
34  			long duration = end-start;
35  			return queryId+" execution time : "+duration+" MegaCycles, or "+quotaService.convertMegacyclesToCpuSeconds(duration)+ " seconds";
36  		}
37  	}
38  	
39  	/**
40  	 * Map of inspected queries
41  	 */
42  	private Map<QueryStatement<?, ?>, QueryInfos> queries = new WeakHashMap<QueryStatement<?,?>, QueryInfos>();
43  
44      QuotaService quotaService = QuotaServiceFactory.getQuotaService();
45  	/**
46  	 * A query changes it status
47  	 */
48  	@Override
49  	public void propertyChange(PropertyChangeEvent evt) {
50  		if (QueryStatement.STATE_PROPERTY.equals(evt.getPropertyName())) {
51  			if (evt.getSource() instanceof QueryStatement) {
52  				State newState = (State) evt.getNewValue();
53  				switch (newState) {
54  				case INITIAL:
55  				case MATCHING:
56  					addQuery((QueryStatement<?, ?>) evt.getSource());
57  					break;
58  				case SORTING:
59  					break;
60  				case EXECUTED:
61  					removeQuery((QueryStatement<?, ?>) evt.getSource());
62  				}
63  			}
64  		}
65  
66  	}
67  
68  	private void removeQuery(QueryStatement<?, ?> source) {
69  		QueryInfos infos = queries.remove(source);
70  		if(infos!=null) {
71  			logger.info(infos.terminate());
72  		}
73  	}
74  
75  	private void addQuery(QueryStatement<?, ?> query) {
76  		queries.put(query, new QueryInfos(query.getId()));
77  	}
