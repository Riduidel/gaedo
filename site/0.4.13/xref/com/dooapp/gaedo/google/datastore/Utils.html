19  	 * 
20  	 * @param type
21  	 * @param property
22  	 * @return
23  	 */
24  	public static Object getCastedProperty(Class<?> type, Object property) {
25  		if (type.isPrimitive()) {
26  			if (Integer.TYPE.equals(type)) {
27  				return ((Number) property).intValue();
28  			} else if (Long.TYPE.equals(type)) {
29  				return ((Number) property).longValue();
30  			} else if (Float.TYPE.equals(type)) {
31  				return ((Number) property).floatValue();
32  			} else if (Double.TYPE.equals(type)) {
33  				return ((Number) property).doubleValue();
34  			} else if (Byte.TYPE.equals(type)) {
35  				return ((Number) property).byteValue();
36  			} else if (Short.TYPE.equals(type)) {
37  				return ((Number) property).shortValue();
38  			} else if (Boolean.TYPE.equals(type)) {
39  				return property;
40  			}
41  		}
42  		return type.cast(property);
43  	}
