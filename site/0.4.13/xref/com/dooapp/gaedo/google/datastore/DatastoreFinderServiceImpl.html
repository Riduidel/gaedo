153 		this.hierarchyManager = hierarchyManager;

155 		// If there is an issue with parent field, it should had been seen by id manager
156 		List<Property> parents = AnnotationsFinder.findAll(provider.get(containedClass), Annotations.PARENT);

169 			logger.info("cache started successfully !");
170 		} catch (CacheException e) {

180 	 * Create an object of the given type in the datastore, and return a
181 	 * reference to the stored object (may not be == to toCreate, but will be
182 	 * {@link Object#equals(Object)})
183 	 */
184 	@Override
185 	public DataType create(DataType toCreate) {
186 		if (logger.isLoggable(Level.FINE)) {
187 			logger.fine("creating object " + toCreate);
188 		}
189 		if(parentField!=null) {
190 			// Ensure parent exists. For that, we check if this parent has a key. if it is not the case, the parent has never been saved.
191 			DatastoreFinderService parentService = (DatastoreFinderService) repository.get(parentField.getType());
192 			Object parentObject = parentField.get(toCreate);
193 			if(parentObject==null) {
194 				throw new ImpossibleToSaveObjectWithNullParentException(toCreate, containedClass, parentField);
195 			}
196 			if(!parentService.getIdManager().hasKey(parentService.getKind(), parentObject)) {
197 				parentObject = parentService.create(parentObject);
198 				parentField.set(toCreate, parentObject);
199 			}
200 		}
201 		Key key = datastore.put(getEntity(toCreate, false));
202 		if (logger.isLoggable(Level.FINE)) {
203 			logger.fine("it has key " + key);
204 		}
205 		try {
206 			return getObjectFromKey(key);
207 		} finally {
208 			// Yup^, getEntity,when given the false boolean value, puts the
209 			// object in that infamous map
210 			// notice we remove it in the finally block, to ensure it is done
211 			// after all elements have been created
212 			removeAccessed(key);
213 		}
214 	}

223 	/**
224 	 * When performing a deletion, we start by getting all child nodes to add them to the delete list (cascade delete indeed does not seems to be natively
225 	 * integrated to GAE - look at http://code.google.com/intl/fr-FR/appengine/docs/java/datastore/relationships.html#Dependent_Children_and_Cascading_Deletes
226 	 * for the scary details).
227 	 */
228 	@Override
229 	public void delete(DataType toDelete) {
230 		if (logger.isLoggable(Level.FINE)) {
231 			logger.fine("deleting object " + toDelete);
232 		}
233 		Key key = getEntity(toDelete, true).getKey();
234 		Collection<Key> keysToDelete = new LinkedList<Key>();
235 		keysToDelete.add(key);
236 		// Delete all children : both automatic ones (like collection indexes) and manual ones (object declared as children)
237 		Query toExecute = new Query(key);
238 		toExecute = toExecute.setKeysOnly();
239 		PreparedQuery prepared = datastore.prepare(toExecute);
240 		Iterable<Entity> result = prepared.asIterable();
241 		for(Entity e : result) {
242 			keysToDelete.add(e.getKey());
243 		}
244 		datastore.delete(keysToDelete);
245 		if (logger.isLoggable(Level.FINE)) {
246 			logger.fine("object " + toDelete + " has been deleted");
247 		}
248 		cache.put(KEY_TO_EVICT, key);
249 	}

267 	 * Create an entity from the input object data. Notice that if the associated data class has defined a parent element,
268 	 * we will first ensure this parent key exist in order to use it to create entity
269 	 * 
270 	 * @param data
271 	 * @param withKey
272 	 *            when set to true, the key value will be rebuilt. Elsewhere, it
273 	 *            will be forgotten
274 	 * @return
275 	 */
276 	private Entity getEntity(DataType data, boolean withKey) {
277 		if (logger.isLoggable(Level.FINER)) {
278 			logger.finer("building entity (with key ? " + withKey
279 					+ ") from object " + data);
280 		}
281 		Entity returned;
282 		Key key;
283 		// key needs to be created. For easier code, we prefer to generate it
284 		// before. Notice key may have been created before if object is parent of another one
285 		if (!withKey) {
286 			if(!idManager.hasKey(getKind(), data))
287 				idManager.createKey(getKind(), data);
288 		}
289 		// Then, build the key from the known id
290 		key = idManager.getKey(getKind(), data);
291 		if (logger.isLoggable(Level.FINER)) {
292 			logger.finer("object uses key " + key);
293 		}
294 		// Avoid stack overflows
295 		objectsBeingAccessed.put(key, new WeakReference<DataType>(data));
296 		// And create an entity with that key
297 		returned = new Entity(key);
298 		// A class field is generated to store live class (in order to be able
299 		// to instanciate object using that class, rather than containedClass
300 		returned
301 				.setProperty(CLASS_PROPERTY, data.getClass().getCanonicalName());
302 		if (logger.isLoggable(Level.FINER)) {
303 			logger.finer("putting all data in the correct fields for " + data);
304 		}
305 		map(returned, data, new EntityFiller<DataType, InformerType>(
306 				repository, datastore));
307 		// Finally, if class supports migration, persists the field used as key
308 		if(migrator!=null) {
309 			String liveVersionFieldName = migrator.getLiveVersionFieldName();
310 			Property liveProperty = getInformer().get(liveVersionFieldName).getField();
311 			returned.setProperty(migrator.getPersistedVersionFieldName(), Utils.getDatastoreFieldName(liveProperty));
312 		}
313 		if (logger.isLoggable(Level.FINER)) {
314 			logger.finer("all data of " + data + " has been put");
315 		}
316 		return returned;
317 	}

342 	 * Get object associated to given entity
343 	 * 
344 	 * @param entity
345 	 * @return
346 	 */
347 	public DataType getObject(Entity entity) {
348 		if(!getKind().equals(entity.getKind())) {
349 			// Try to lookup parents
350 			Key input = entity.getKey();
351 			Key current = input;
352 			while(current!=null && !(getKind().equals(current.getKind()))) {
353 				current = current.getParent();
354 			}
355 			if(current==null) {
356 				throw new UnableToCreateObjectDueToBadKeyException(input, containedClass);
357 			}
358 			try {
359 				entity = datastore.get(current);
360 			} catch (EntityNotFoundException e) {
361 				throw new UnableToCreateObjectException(e, containedClass);
362 			}
363 		}
364 		// Load object from source entity
365 		try {
366 			if (logger.isLoggable(Level.FINER)) {
367 				logger.finer("getting object associated to  " + entity);
368 			}
369 			// Do some entity check to see if a migration is required
370 			if(migrator!=null) {
371 				String storedVersionField = entity.getProperty(migrator.getPersistedVersionFieldName()).toString();
372 				Object storedVersion = entity.getProperty(storedVersionField);
373 				Object liveVersion = migrator.getCurrentVersion();
374 				if(!storedVersion.equals(liveVersion)) {
375 					entity = migrator.migrate(this, entity, storedVersion, liveVersion);
376 				}
377 			}
378 			DataType returned;
379 			try {
380 				String dynamicClass = (String) entity
381 						.getProperty(CLASS_PROPERTY);
382 				if (logger.isLoggable(Level.FINER)) {
383 					logger.finer("object should be a " + dynamicClass);
384 				}
385 				Class<?> instanciated = Class.forName(dynamicClass);
386 				returned = (DataType) instanciated.newInstance();
387 				objectsBeingAccessed.put(entity.getKey(),
388 						new WeakReference<DataType>(returned));
389 			} catch (Exception e) {
390 				throw new UnableToCreateObjectException(e, containedClass);
391 			}
392 			if (logger.isLoggable(Level.FINER)) {
393 				logger.finer("mapping fields");
394 			}
395 			// Set all fields values
396 			map(entity, returned, new ObjectFiller<DataType, InformerType>(
397 					repository, datastore));
398 			if (logger.isLoggable(Level.FINER)) {
399 				logger.finer("setting key");
400 			}
401 			// Now set id value
402 			idManager.setKey(getKind(), entity.getKey(), returned);
403 			if (logger.isLoggable(Level.FINER)) {
404 				logger.finer("object is usable");
405 			}
406 			return returned;
407 		} finally {
408 			objectsBeingAccessed.remove(entity.getKey());
409 		}
410 	}

427 	/**
428 	 * Create the fields map used to store infos from this class.
429 	 * 
430 	 * @param clazz
431 	 *            class for which we want to have all fields
432 	 * @return a Map linking unique-made names for fields to the Field objects
433 	 */
434 	private Map<String, Property> createFields(Class<?> clazz) {
435 		Map<String, Property> returned = new TreeMap<String, Property>();
436 		if (!clazz.equals(Object.class)) {
437 			if (!classes.containsKey(clazz)) {
438 				Map<String, Property> fields = new TreeMap<String, Property>();
439 				for (Property f : PropertyProviderUtils.getAllProperties(propertyProvider, clazz)) {
440 					// Remove transient fields (as of issue #19)
441 					if (!f.hasModifier(Modifier.TRANSIENT) && !f.hasModifier(Modifier.STATIC))
442 						fields.put(Utils.getDatastoreFieldName(f), f);
443 				}
444 				classes.put(clazz, fields);
445 			}
446 			returned.putAll(classes.get(clazz));
447 			returned.putAll(createFields(clazz.getSuperclass()));
448 		}
449 		return returned;
450 	}

587 	public DataType getObjectFromKey(Key key) {
588 		if (objectsBeingAccessed.containsKey(key)) {
589 			DataType returned = objectsBeingAccessed.get(key).get();
590 			if (returned == null) {
591 				objectsBeingAccessed.remove(key);
592 			} else {
593 				return returned;
594 			}
595 		}
596 		return loadObject(key);
