74  	public void loadService() throws Exception {
75  		serviceRrepository = new SimpleServiceRepository();
76  		provider = new FieldBackedPropertyProvider();
77  		locator = new CumulativeFieldInformerLocator();
78  		locator.add(new BasicFieldInformerLocator());
79  		locator.add(new ServiceBackedFieldLocator(serviceRrepository));
80  		locator.add(new LazyInterfaceInformerLocator());
81  		reflectiveFactory = new ReflectionBackedInformerFactory(
82  				locator, provider);
83  		proxyInformerFactory = new ProxyBackedInformerFactory(
84  				reflectiveFactory);
85  		
86  		graph = createGraph(graphProvider);
87  		
88  		// Now add some services
89  		tagService = createServiceFor(Tag.class, TagInformer.class, strategy );
90  		postService = createServiceFor(Post.class, PostInformer.class, strategy);
91  		postSubService= createServiceFor(PostSubClass.class, PostSubClassInformer.class, strategy);
92  		userService = createServiceFor(User.class, UserInformer.class, strategy);
93  		themeService = createServiceFor(Theme.class, ThemeInformer.class, strategy);
94  	}
