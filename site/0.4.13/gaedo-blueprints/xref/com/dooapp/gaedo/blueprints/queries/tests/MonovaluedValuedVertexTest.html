44  	 * @param examined

46  	 * @see com.dooapp.gaedo.blueprints.queries.tests.VertexTest#matches(com.tinkerpop.blueprints.pgm.Vertex)
47  	 */
48  	@Override
49  	public boolean matches(Vertex examined) {
50  		// Navigates to the first target edge and perform etest when reached
51  		Vertex currentVertex = examined;
52  		Property finalProperty = null;
53  		// Counting path length allows us to check if we expect a null value
54  		int currentPathLength = 0;
55  		for(Property currentProperty : path) {
56  			Iterator<Edge> edges = currentVertex.getOutEdges(GraphUtils.getEdgeNameFor(currentProperty)).iterator();
57  			if(edges.hasNext()) {
58  				currentVertex = edges.next().getInVertex();
59  			} else {
60  				return false;
61  			}
62  			finalProperty = currentProperty;
63  			currentPathLength++;
64  		}
65  		if(finalProperty==null) {
66  			return false;
67  		} else {
68  			return matchesVertex(currentVertex, finalProperty);
69  		}
70  	}

79  		if(getRepository().containsKey(expected.getClass())) {
