65  	public static String getEdgeNameFor(Property p) {
66  		if (p.getAnnotation(GraphProperty.class) != null) {
67  			GraphProperty graph = p.getAnnotation(GraphProperty.class);
68  			// Test added to avoid default value (which defaults name to "")
69  			if (graph.name() != null && graph.name().trim().length() > 0)
70  				return graph.name();

108 	 * Create an object instance from a literal vertex compatible with this
109 	 * service contained class
110 	 * 
111 	 * @param driver
112 	 *            driver used to load data
113 	 * @param strategy
114 	 *            used graph mapping strategy
115 	 * @param classLoader
116 	 *            class loader used to find class
117 	 * @param key
118 	 *            vertex containing object id
119 	 * @param repository
120 	 *            service repository, used to disambiguate subclass of literal
121 	 *            and managed class
122 	 * @param objectsBeingAccessed
123 	 * @param property
124 	 *            property used to navigate to this value. it allows
125 	 *            disambiguation for literal values (which may be linked to more
126 	 *            than one type, the typical example being a saved float, say
127 	 *            "3.0", which may also be refered as the string "3.0").
128 	 * @return a fresh instance, with only id set
129 	 */
130 	public static Object createInstance(GraphDatabaseDriver driver, GraphMappingStrategy strategy, ClassLoader classLoader, Vertex key, Class<?> defaultType,
131 					ServiceRepository repository, Map<String, Object> objectsBeingAccessed) {
132 		String effectiveType = null;
133 		Kind kind = getKindOf(key);
134 		/*
135 		 * One literal node may be used according to different types. To
136 		 * disambiguate, we check if effective type matches default one. If not
137 		 * (typically type returns string and user wants number), prefer default
138 		 * type.
139 		 */
140 		try {
141 			effectiveType = driver.getEffectiveType(key);
142 		} catch (UnableToGetVertexTypeException untypedVertex) {
143 			try {
144 				// Don't remember the reason of that mess
145 				if (!Collection.class.isAssignableFrom(defaultType) && !defaultType.isAssignableFrom(Class.forName(effectiveType))) {
146 					effectiveType = defaultType.getName();
147 				}
148 			} catch (Exception unableToLoadClass) {
149 				// nothing to do : we use effective type - or try to
150 			}
151 			if (effectiveType == null) {
152 				// First alternative is here for untyped strings in uris nodes (like uris themselves when treated as strings)
153 				// Second alternative is there for cases when we try to load a collection of untyped thingies
154 				if (String.class.isAssignableFrom(defaultType) || Collection.class.isAssignableFrom(defaultType))
155 					effectiveType = GraphMappingStrategy.STRING_TYPE;
156 
157 			}
158 		}
159 		if (classLoader == null) {
160 			throw new UnspecifiedClassLoader();
161 		}
162 		try {
163 			if (Literals.containsKey(classLoader, effectiveType) && !repository.containsKey(effectiveType)) {
164 				LiteralTransformer transformer = Literals.get(classLoader, effectiveType);
165 				return transformer.loadObject(driver, classLoader, effectiveType, key);
166 			} else {
167 				Class<?> type = classLoader.loadClass(effectiveType);
168 				if (Tuples.containsKey(type) && !repository.containsKey(type)) {
169 					// Tuples are handled the object way (easier, but more
170 					// dangerous
171 					TupleTransformer transformer = Tuples.get(type);
172 					return transformer.loadObject(driver, strategy, classLoader, type, key, repository, objectsBeingAccessed);
173 				} else {
174 					return type.newInstance();
175 				}
176 			}
177 		} catch (Exception e) {
178 			throw new UnableToCreateException(effectiveType, e);
179 		}
180 	}

196 		Class<? extends Object> valueClass = value.getClass();
197 		if (repository.containsKey(valueClass)) {
198 			AbstractBluePrintsBackedFinderService<IndexableGraph, DataType, ?> service = (AbstractBluePrintsBackedFinderService<IndexableGraph, DataType, ?>) repository
199 							.get(valueClass);
200 			// All ids are string, don't worry about it
201 			return service.getIdOf(value).toString();
202 		} else if (Literals.containsKey(valueClass)) {
203 			return getIdOfLiteral(valueClass, null, value);
204 		} else if (Tuples.containsKey(valueClass)) {
205 			return getIdOfTuple(repository, valueClass, value);
206 		} else {
207 			throw new ImpossibleToGetIdOfUnknownType(valueClass);
208 		}

287 		Set<CascadeType> returned = new HashSet<CascadeType>();
288 		returned.addAll(Arrays.asList(cascade));

315 	/**
316 	 * Find all contexts in given edge by looking, in
317 	 * {@link GraphSail#CONTEXT_PROP} property, what are the contexts. These
318 	 * contexts are extracted by iteratively calling {@link #CONTEXTS_MATCHER}
319 	 * and {@link Matcher#find(int)} method.
320 	 * 
321 	 * @param edge
322 	 *            input edge
323 	 * @return collection of declared contexts.
324 	 */
325 	public static Collection<String> getContextsOf(Edge edge) {
326 		String contextsString = edge.getProperty(GraphSail.CONTEXT_PROP).toString();
327 		Matcher matcher = CONTEXTS_MATCHER.matcher(contextsString);
328 		Collection<String> output = new LinkedList<String>();
329 		int character = 0;
330 		while (matcher.find(character)) {
331 			if (GraphSail.NULL_CONTEXT_NATIVE.equals(matcher.group(1))) {
332 				// the null context is a low-level view. It should be associated
333 				// with "no named graph" (that's to say an empty collection).
334 				return output;
335 			} else if (matcher.group(1).startsWith(GraphSail.URI_PREFIX + "")) {
336 				output.add(matcher.group(2));
337 			}
338 			character = matcher.end();
339 		}
340 		return output;
341 	}
