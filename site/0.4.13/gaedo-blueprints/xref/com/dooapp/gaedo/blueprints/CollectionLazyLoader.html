31  	/**
32  	 * Comparator used to sort the Edges linking an object with the the elements of a Collection.
33  	 * This is used to maintain order in ordered collections like List.
34  	 */
35  	private static Comparator<Edge> COLLECTION_EDGE_COMPARATOR = new Comparator<Edge>() {
36  		@Override
37  		public int compare(Edge o1, Edge o2) {
38  			Integer o1Idx = (Integer) o1.getProperty(Properties.collection_index.name());
39  			Integer o2Idx = (Integer) o2.getProperty(Properties.collection_index.name());
40  			
41  			if(null == o1Idx || null == o2Idx)
42  				throw new UnableToSortException("An edge in an ordered collection does not have an index, so we don't know where to put it!");
43  			
44  			return o1Idx.compareTo(o2Idx);
45  		}
46  	};

62  		if(!loaded) {

64  		}
65  		if(method.getDeclaringClass().equals(WriteReplaceable.class)) {

72  	@SuppressWarnings("unchecked")
73  	public void loadCollection(Collection collection, Map<String, Object> objectsBeingAccessed) {
74  		try {
75  			// Use the magic order property to try to put the elements in the correct order (if the property is there)
76  			List<Edge> edges = new LinkedList<Edge>();
77  			boolean needToSort = false;
78  			for(Edge e : rootVertex.getOutEdges(edgeName)) {
79  				edges.add(e);
80  				if(e.getProperty(Properties.collection_index.name()) != null)
81  					needToSort = true;
82  			}
83  			if(needToSort)
84  				Collections.sort(edges, COLLECTION_EDGE_COMPARATOR);
85  			
86  			// Now that everything is in order, we can load the real collection
87  			for(Edge e : edges) {
88  				Vertex value = e.getInVertex();
89  				try {
90  					Object temporaryValue = GraphUtils.createInstance(driver, strategy, classLoader, value, property.getType(), repository, objectsBeingAccessed);
91  					if(repository.containsKey(temporaryValue.getClass())) {
92  						FinderCrudService service = repository.get(temporaryValue.getClass());
93  						if (service instanceof AbstractBluePrintsBackedFinderService) {
94  							AbstractBluePrintsBackedFinderService<?, ?, ?> blueprints= (AbstractBluePrintsBackedFinderService<?, ?, ?>) service;
95  							collection.add(blueprints.loadObject(value, objectsBeingAccessed));
96  						}
97  					} else {
98  						// Instance should be OK, as createinstance should support everything getVertexForBasicObject supports
99  						collection.add(temporaryValue);
100 					}
101 				} catch(UnableToCreateException ex) {
102 					if (logger.isLoggable(Level.WARNING)) {
103 						logger.log(Level.WARNING, "we failed to load value associated with vertex "+GraphUtils.toString(value), ex);
104 					}
105 				}
106 			}
107 		} finally {
108 			loaded = true;
109 		}
110 	}
