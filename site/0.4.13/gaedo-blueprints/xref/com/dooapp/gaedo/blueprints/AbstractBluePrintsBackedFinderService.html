255 	 * Local delete implementation
256 	 * 
257 	 * @param toDelete
258 	 */
259 	private void doDelete(DataType toDelete, Map<String, Object> objectsBeingAccessed) {
260 		String vertexId = getIdVertexId(toDelete, false /*
261 																	 * no id
262 																	 * generation
263 																	 * on delete
264 																	 */);
265 		Vertex objectVertex = loadVertexFor(vertexId, toDelete.getClass().getName());
266 		if (objectVertex != null) {
267 			Map<Property, Collection<CascadeType>> containedProperties = strategy.getContainedProperties(toDelete, objectVertex, CascadeType.REMOVE);
268 			persister.performDelete(this, vertexId, objectVertex, containedClass, containedProperties, toDelete, CascadeType.REMOVE, objectsBeingAccessed);
269 		}

272 	/**
273 	 * Delete an out edge vertex. Those are vertex corresponding to properties.
274 	 * 
275 	 * @param objectVertex
276 	 *            source object vertex, used for debugging purpose only
277 	 * @param valueVertex
278 	 *            value vertex to remove
279 	 * @param value
280 	 *            object value
281 	 */
282 	<Type> void deleteOutEdgeVertex(Vertex objectVertex, Vertex valueVertex, Type value, Map<String, Object> objectsBeingUpdated) {
283 		// Delete vertex and other associated ones, only if they have no
284 		// other input links (elsewhere delete is silently ignored)
285 		if (valueVertex.getInEdges().iterator().hasNext()) {
286 			// There are incoming edges to that vertex. Do nothing but log
287 			// it
288 			if (logger.isLoggable(Level.FINE)) {
289 				logger.log(Level.FINE,
290 								"while deleting " + GraphUtils.toString(objectVertex) + "" + " we tried to delete " + GraphUtils.toString(valueVertex)
291 												+ "" + " which has other incoming edges, so we didn't deleted it");
292 			}
293 		} else {
294 			// OK, time to delete value vertex. Is it a managed node ?
295 			if (repository.containsKey(value.getClass())) {
296 				FinderCrudService<Type, ?> finderCrudService = (FinderCrudService<Type, ?>) repository.get(value.getClass());
297 				if (finderCrudService instanceof AbstractBluePrintsBackedFinderService) {
298 					((AbstractBluePrintsBackedFinderService<?, Type, ?>) finderCrudService).doDelete(value, objectsBeingUpdated);
299 				} else {
300 					throw new IncompatibleServiceException(finderCrudService, value.getClass());
301 				}
302 			} else {
303 				// Literal nodes can be deleted without any trouble
304 				database.removeVertex(valueVertex);
305 			}
306 		}
307 	}

354 		return new TransactionalOperation<DataType, DataType, InformerType>(this) {
355 
356 			@Override
357 			protected DataType doPerform() {
358 				return doUpdate(toUpdate, CascadeType.MERGE, new TreeMap<String, Object>());
359 			}
360 		}.perform();
361 	}

363 	/**
364 	 * here is a trick : we want id generation to happen only on first persist
365 	 * (that's to say on call to #create), but not on subsequent ones. So, as
366 	 * first call uses CascadeType.PERSIST and others uses CascadeType.MERGE, we
367 	 * can use that indication to separate them. It has the unfortunate
368 	 * inconvenient to force us to use only PERSIST during #create
369 	 * 
370 	 * @param toUpdate
371 	 *            object to update
372 	 * @param cascade
373 	 *            type. As mentionned upper, beware to value used !
374 	 * @param treeMap
375 	 *            map of objects already used
376 	 */
377 	private DataType doUpdate(DataType toUpdate, CascadeType cascade, Map<String, Object> treeMap) {
378 		boolean generatesId = strategy.isIdGenerationRequired() ? (CascadeType.PERSIST == cascade) : false;
379 		String objectVertexId = getIdVertexId(toUpdate, generatesId);
380 		Vertex objectVertex = loadVertexFor(objectVertexId, toUpdate.getClass().getName());
381 		return (DataType) persister.performUpdate(this, objectVertexId, objectVertex, toUpdate.getClass(), strategy.getContainedProperties(toUpdate, objectVertex, cascade), toUpdate, cascade,
382 						treeMap);
383 	}

385 	/**
386 	 * Get vertex associated to value. If object is managed by a service, we ask
387 	 * this service the value
388 	 * 
389 	 * @param value
390 	 *            value we want the vertex for
391 	 * @param cascade
392 	 *            used cascade type, can be either {@link CascadeType#PERSIST}
393 	 *            or {@link CascadeType#MERGE}
394 	 * @param objectsBeingUpdated
395 	 *            map of objects currently being updated, it avoid some loops
396 	 *            during update, but is absolutely NOT a persistent cache
397 	 * @return vertex for given value. May be null in soem rare cases (typically obtaining a vertex that doesn't exist yet for deleting it)
398 	 */
399 	public Vertex getVertexFor(Object value, CascadeType cascade, Map<String, Object> objectsBeingUpdated) {
400 		boolean allowIdGeneration = CascadeType.PERSIST.equals(cascade) || CascadeType.MERGE.equals(cascade);
401 		// Here we suppose the service is the right one for the job (which may
402 		// not be the case)
403 		if (containedClass.isInstance(value)) {
404 			Vertex returned = getIdVertexFor(containedClass.cast(value), allowIdGeneration);
405 			if (returned == null) {
406 				doUpdate(containedClass.cast(value), cascade, objectsBeingUpdated);
407 				returned = getIdVertexFor(containedClass.cast(value), allowIdGeneration);
408 			} else {
409 				// vertex already exist, but maybe object needs an update
410 				if (CascadeType.PERSIST == cascade || CascadeType.MERGE == cascade) {
411 					doUpdate(containedClass.cast(value), cascade, objectsBeingUpdated);
412 				}
413 			}
414 			return returned;
415 		}
416 		Class<? extends Object> valueClass = value.getClass();
417 		if (repository.containsKey(valueClass)) {
418 			FinderCrudService service = repository.get(valueClass);
419 			if (service instanceof AbstractBluePrintsBackedFinderService) {
420 				return ((AbstractBluePrintsBackedFinderService) service).getVertexFor(value, cascade, objectsBeingUpdated);
421 			} else {
422 				throw new IncompatibleServiceException(service, valueClass);
423 			}
424 		} else if (Literals.containsKey(valueClass)) {
425 			return getVertexForLiteral(value);
426 		} else if (Tuples.containsKey(valueClass)) {
427 			return getVertexForTuple(value, objectsBeingUpdated);
428 		} else {
429 			/*
430 			 * // OK, we will persist this object by ourselves, which is really
431 			 * error-prone, but we do we have any other solution ? // But notice
432 			 * object is by design consderie Vertex objectVertex =
433 			 * objectVertex.setProperty(Properties.vertexId.name(),
434 			 * getIdVertexId(toUpdate));
435 			 * objectVertex.setProperty(Properties.kind.name(),
436 			 * Kind.managed.name());
437 			 * objectVertex.setProperty(Properties.type.name(),
438 			 * toUpdate.getClass().getName());
439 			 */
440 			throw new ObjectIsNotARealLiteralException(value, valueClass);
441 
442 		}
443 	}

524 	/**
525 	 * we only consider first id element
526 	 * 
527 	 * @param id
528 	 *            collection of id
529 	 * @return object which has as vertexId the given property
530 	 * @see com.dooapp.gaedo.finders.id.IdBasedService#findById(java.lang.Object[])
531 	 */
532 	@Override
533 	public DataType findById(final Object... id) {
534 		// make sure entered type is a valid one
535 		String vertexIdValue = strategy.getAsId(id[0]);
536 		Vertex rootVertex = loadVertexFor(vertexIdValue, containedClass.getName());
537 		if (rootVertex == null) {
538             // root vertex couldn't be found directly, mostly due to
539             // https://github.com/Riduidel/gaedo/issues/11
540             // So perform the longer (but always working) query
541             return find().matching(new QueryBuilder<InformerType>() {
542 
543                 @Override
544                 public QueryExpression createMatchingExpression(InformerType informer) {
545                     Collection<QueryExpression> ands = new LinkedList<QueryExpression>();
546                     int index=0;
547                     for(Property idProperty : strategy.getIdProperties()) {
548                         ands.add(informer.get(idProperty.getName()).equalsTo(id[index++]));
549                     }
550                     return Expressions.and(ands.toArray(new QueryExpression[ands.size()]));
551                 }
552             }).getFirst();
553 		} else {
554 			// root vertex can be directly found ! so load it immediatly
555 			return loadObject(vertexIdValue, new TreeMap<String, Object>());
556 		}
557 	}

594 	/**
595 	 * Set id of object, and try to assign that object a vertex.
596 	 * 
597 	 * @param value
598 	 * @param id
599 	 * @return
600 	 * @see com.dooapp.gaedo.finders.id.IdBasedService#assignId(java.lang.Object,
601 	 *      java.lang.Object[])
602 	 */
603 	@Override
604 	public boolean assignId(final DataType value, Object... id) {
605 		/*
606 		 * We first make sure object is an instance of containedClass. This way,
607 		 * we can then use value class to create id vertex
608 		 */
609 		if (containedClass.isInstance(value)) {
610 			strategy.assignId(value, id);
611 			if (getIdVertexFor(value, false /*
612 											 * no id generation when assigning
613 											 * an id !
614 											 */) == null) {
615 				try {
