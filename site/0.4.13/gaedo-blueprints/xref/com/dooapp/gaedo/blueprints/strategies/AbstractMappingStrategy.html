99  	public void generateIdFor(AbstractBluePrintsBackedFinderService<?, DataType, ?> service, DataType toCreate) {
100 		IdGenerator generator = null;
101 		Class<?> objectType = Utils.maybeObjectify(idProperty.getType());
102 		if (Long.class.isAssignableFrom(objectType)) {
103 			generator = new LongGenerator(service, idProperty);
104 		} else if (Integer.class.isAssignableFrom(objectType)) {
105 			generator = new IntegerGenerator(service, idProperty);
106 		} else if (String.class.isAssignableFrom(objectType)) {
107 			generator = new StringGenerator(service, idProperty);
108 		} else {
109 			throw new UnsupportedIdTypeException(objectType + " can't be used as id : we don't know how to generate its values !");
110 		}
111 		generator.generateIdFor(toCreate);
112 	}
