23  			Map<String, Object> sortParams) {

25  		// Now explore tree to build a QueryBuilder instance

27  		for(Map.Entry<String, Object> entry : valuesAsTree.entrySet()) {
28  			buildSortingExpression(combinator, service.getInformer(), (Map<String, Object>) entry.getValue());
29  		}
30  		return new SortingBuilder<InformerType>() {
31

34  					InformerType informer) {

36  			}

38  	}

40  	private void buildSortingExpression(SortingExpressionImpl parent,
41  			FieldInformer fieldInformer, Map<String, Object> value) {
42  		for(Map.Entry<String, Object> entry : value.entrySet()) {
43  			if(fieldInformer instanceof Informer) {
44  				Informer beanInformer = (Informer) fieldInformer;
45  				try {
46  					FieldInformer usedInformer = beanInformer.get(entry.getKey());
47  					parent.add(usedInformer, SortingExpression.Direction.valueOf(entry.getValue().toString()));
48  				} catch(Exception e) {
49  					logger.log(Level.FINE, "unable to create a sorting expression from field name "+entry.getKey()+" in class "+beanInformer.toString(), e);
50  				}
51  			}
52  		}
53  	}
