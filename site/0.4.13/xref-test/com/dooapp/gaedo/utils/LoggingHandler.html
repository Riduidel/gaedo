13  /**
14   * A logging invocation handler allowing automatic unwrapping and a log for each test
15   * @author ndx
16   *
17   */
18  public class LoggingHandler implements InvocationHandler {
19  	private static final Logger logger = Logger.getLogger(LoggingHandler.class.getName());
20  	
21  	private Object source;
22  
23  	public LoggingHandler(Object source) {
24  		this.source = source;
25  	}
26  
27  	@Override
28  	public Object invoke(Object proxy, Method method, Object[] args) throws Throwable {
29  		StopWatch watch = new StopWatch();
30  		watch.start();
31  		try {
32  			// Unwrap potentially wrapped objects
33  			Object returned = method.invoke(source, unwrap(args));
34  			// java.* methods aren't decorated
35  			if(returned!=null && method.getReturnType().isInterface() && !method.getDeclaringClass().getCanonicalName().startsWith("java")) {
36  				returned = Proxy.newProxyInstance(getClass().getClassLoader(), new Class[] {method.getReturnType()}, new LoggingHandler(returned));
37  			}
38  			return returned;
39  		} finally {
40  			watch.stop();
41  			if (logger.isLoggable(Level.INFO)) {
42  				logger.info(new ToStringBuilder(source, ToStringStyle.MULTI_LINE_STYLE).append("method", method).append("arguments", args).append("duration", watch.toString()).toString());
43  			}
44  		}
45  	}
46  
47  	
48  	private Object[] unwrap(Object[] args) {
49  		if(args==null)
50  			return args;
51  		Object[] returned = new Object[args.length];
52  		for (int index = 0; index < returned.length; index++) {
53  			if(args[index] instanceof Proxy) {
54  				InvocationHandler handler = Proxy.getInvocationHandler(args[index]);
55  				if(handler instanceof LoggingHandler) {
56  					returned[index] = ((LoggingHandler) handler).source;
57  				}
58  			} else {
59  				returned[index] = args[index];
60  			}
61  		}
62  		return returned;
63  	}
