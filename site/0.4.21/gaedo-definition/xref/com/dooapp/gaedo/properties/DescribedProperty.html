40  	 * @param descriptor Property descriptor used to provide infos on which methods to call.
41  	 * Due to the very nature of PropertyDescriptor and its Soft/Weak references to write and read methods, all infos are copied here (we hope PropertyDescriptor is fresh enough
42  	 * to not have been GCed).
43  	 * @param declaringClass
44  	 */
45  	public DescribedProperty(PropertyDescriptor descriptor, Class<?> declaringClass) {
46  		this.declaringClass = declaringClass;
47  		this.readMethod = descriptor.getReadMethod();
48  		this.writeMethod = descriptor.getWriteMethod();
49  		this.methods = new Method[] {readMethod, writeMethod};
50  		type = descriptor.getPropertyType();
51  		name = descriptor.getName();
52  		descriptorString = descriptor.toString();
53  	}

67  	@Override

69  		for(Method m : methods) {
70  			if(m.getAnnotation(annotationClass)!=null)
71  				return m.getAnnotation(annotationClass);
72  		}
73  		return null;

76  	@Override
77  	public Collection<? extends Annotation> getAnnotations() {
78  		Collection<Annotation> returned = new HashSet<Annotation>();
79  		for(Method m : methods) {
80  			// beware : for read-only/write-only fields, one o the two methods is null
81  			if(m!=null) {
82  				returned.addAll(Arrays.asList(m.getAnnotations()));
83  			}
84  		}
85  		return returned;
