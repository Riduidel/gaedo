39  		CHILD(Arrays

42  		private Annotations(Collection<String> names) {
43  			this.annotationsNames = Collections.unmodifiableCollection(names);
44  		}
45  
46  		private Collection<String> annotationsNames;
47  
48  		/**
49  		 * 
50  		 * @param annotations
51  		 * @return true if this particular annotation list has any of the input
52  		 *         annotations
53  		 */
54  		public boolean hasAny(Collection<? extends Annotation> annotations) {
55  			for (Annotation a : annotations) {
56  				if (annotationsNames.contains(a.annotationType().getName()))
57  					return true;
58  			}
59  			return false;
60  		}
61  
62  		/**
63  		 * Check if property has the given annotation type
64  		 * @param p property to check
65  		 * @return true if property contains any of the given annotations
66  		 */
67  		public boolean is(Property p) {
68  			return hasAny(p.getAnnotations());
69  		}
70  	}
71  
72  	/**
73  	 * Find the first field in class that has the id annotation
74  	 * 
75  	 * @param fields
76  	 *            collection of fields to lookup
77  	 * @return null if none found
78  	 */
79  	public static List<Property> findAll(Property[] fields,
80  			Annotations toSupport) {
81  		List<Property> returned = new LinkedList<Property>();
82  		for (Property f : fields) {
83  			// only non static fields are used
84  			if (!f.hasModifier(Modifier.STATIC)) {
85  				Collection<? extends Annotation> annotations = f
86  						.getAnnotations();
87  				if (toSupport.hasAny(annotations)) {
88  					returned.add(f);
89  					break;
90  				}
91  			}
92  		}
93  		return returned;
94  	}
