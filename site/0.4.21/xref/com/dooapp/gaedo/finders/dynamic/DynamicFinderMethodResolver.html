102 	/**
103 	 * Build the QueryExpression, then gently ask {@link Mode#execute(FinderCrudService, QueryExpression, SortingBuilder, Object[])} to perform query.
104 	 * Finally, for findAll, we provide the additionnal feature to put obejcts in the correct Iterable sub-interface associated type
105 	 * @param args
106 	 * @return
107 	 */
108 	public Object call(Object[] args) {
109 		Collection<QueryExpression> resolvedExpressions = new LinkedList<QueryExpression>();
110 		int offset = mode.getOffset();
111 		for(Map.Entry<FieldInformer, Method> entry : queryExpressions) {
112 			int consumableLength = entry.getValue().getParameterTypes().length;
113 			Object[] methodArgs = new Object[consumableLength];
114 			System.arraycopy(args, offset, methodArgs, 0, consumableLength);
115 			offset += consumableLength;
116 			try {
117 				resolvedExpressions.add((QueryExpression) entry.getValue().invoke(entry.getKey(), methodArgs));
118 			} catch (Exception e) {
119 				throw new UnableToBuildQueryExpressionException(entry.getKey(), entry.getValue(), methodArgs, e);
120 			}
121 		}
122 		QueryExpression queryExpression = combinator.create(resolvedExpressions);
123 		// Creates an array of parameters going from 0 to offset
124 		Object[] modeArgs = new Object[mode.getOffset()];
125 		System.arraycopy(args, 0, modeArgs, 0, mode.getOffset());
126 		Object returned = mode.execute(backEnd, queryExpression, sortingExpression, modeArgs, executionId);
127 		// Find all tricky trick
128 		if(mode==Mode.FIND_ALL || mode==Mode.FIND_RANGE) {
129 			// If method return type is not iterable but a sub-interface, put my merlin's cape !
130 			if(!Iterable.class.equals(returnType)) {
131 				Collection toReturn = Utils.generateCollection(returnType, null);
132 				Iterable iterable = (Iterable) returned;
133 				for(Object data : iterable) {
134 					toReturn.add(data);
135 				}
136 				returned = toReturn;
137 			}
138 		}
139 		return returned;
140 	}

142 	/**
143 	 * Check that this MethodResolver uses the same parameters stack than the given Method object
144 	 * @param method input method
145 	 */
146 	public void checkParametersClasses(Method method) {
147 		Collection<String> errors = new LinkedList<String>();
148 		Type[] consumableParameters = method.getGenericParameterTypes();
149 		List<Type> consumableList = new LinkedList<Type>(Arrays.asList(consumableParameters));
150 		// Creates an array of parameters going from 0 to offset
151 		Type[] modeArgs = new Type[mode.getOffset()];
152 		System.arraycopy(consumableParameters, 0, modeArgs, 0, mode.getOffset());
153 		mode.checkParametersClasses(method, modeArgs, this);
154 		int paramIndex = mode.getOffset();
155 		for(Map.Entry<FieldInformer, Method> entry : queryExpressions) {
156 			Class<?>[] parameters = entry.getValue().getParameterTypes();
157 			for(Class<?> p : parameters) {
158 				if(consumableList.size()==0) {
159 					errors.add("there are not enough parameters to match method call "+entry.getKey().toString()+" "+entry.getValue().toGenericString());
160 					continue;
161 				}
162 				Type toCompare = consumableList.remove(0);
163 				if(toCompare instanceof Class<?>) {
164 					Class<?> toCompareClass = (Class<?>) toCompare;
165 					if(toCompareClass.isPrimitive()) {
166 						// Do some magick to retrieve associated class
167 						toCompareClass = Utils.objectify(toCompareClass);
168 					}
169 					if(!p.isAssignableFrom(toCompareClass)) {
170 						errors.add("parameter "+p.getName()+" of method "+entry.getValue().toGenericString()+"cannot use value of type "+toCompareClass.getName());
171 					}
172 				} else {
173 					throw new UnsupportedOperationException(toCompare.getClass().getName()+" not supported in that case. please fill a gaedo-definition bug report");
174 				}
175 			}
176 			paramIndex++;
177 		}
178 		if(errors.size()>0) {
179 			throw new MethodBindingException(method, this, errors);
180 		}
181 	}

215 	/**
216 	 * Check that implemented method return type is compliant with internal code.
217 	 * Notice we can perform some type escalation (from Iterable to Collection, as an example)
218 	 * @param method
219 	 */
220 	private void checkReturnType(Method method) {
221 		Class<?> returnType = method.getReturnType();
222 		switch(mode) {
223 		case COUNT:
224 			if(!(Integer.class.isAssignableFrom(returnType) || Integer.TYPE.isAssignableFrom(returnType) ||
225 					Long.class.isAssignableFrom(returnType) || Long.TYPE.isAssignableFrom(returnType))) {
226 				throw new BadReturnTypeException(method, returnType, Integer.class, int.class, long.class, Long.class);
227 			}
228 			break;
229 		case FIND_ALL:
230 			if(!Iterable.class.isAssignableFrom(returnType)) {
231 				throw new BadReturnTypeException(method, returnType, Iterable.class);
232 			}
233 			break;
234 		case FIND_ONE:
235 			if(!returnType.isAssignableFrom(backEnd.getContainedClass())) {
236 				throw new BadReturnTypeException(method, returnType, backEnd.getContainedClass());
237 			}
238 		}
239 	}
