92  	 * Consume text to populate the sorting expression
93  	 * @category dynamic_method_builder
94  	 */
95  	public String consumeSort(String methodString, String consumableText, Map<String, String> fieldNames,DynamicFinderMethodResolver created) {
96  		boolean found = false;
97  		for(Map.Entry<String, String> fieldEntry: fieldNames.entrySet()) {
98  			if(consumableText.startsWith(fieldEntry.getKey())) {
99  				consumableText = consumableText.substring(fieldEntry.getKey().length());
100 				FieldInformer informer = backEnd.getInformer().get(fieldEntry.getValue());
101 				found = false;
102 				for(SortingExpression.Direction direction : SortingExpression.Direction.values()) {
103 					if(consumableText.startsWith(direction.getText())) {
104 						consumableText = consumableText.substring(direction.getText().length());
105 						created.addSortingExpression(informer, direction);
106 						found = true;
107 					}
108 				}
109 				if(!found) {
110 					throw new UnableToBuildDueToMissingSortingExpressionException(consumableText, methodString, fieldEntry.getKey(),SortingExpression.Direction.values());
111 				}
112 				// Sorting can only be combined with And text
113 				if(consumableText.startsWith(SORT_BY_COMBINATOR)) {
114 					consumableText = consumableText.substring(SORT_BY_COMBINATOR.length());
115 				}
116 				found = true;
117 			}
118 		}
119 		return consumableText;
120 	}

126 	public String consumeFilter(String methodString, String consumableText, Map<String, String> fieldNames,DynamicFinderMethodResolver created) {
127 		boolean found = false;

131 				FieldInformer informer = backEnd.getInformer().get(fieldEntry.getValue());
132 				Map<String, Method> queries = Utils.getUppercasedMap(informer.getClass().getMethods());
133 				found = false;
134 				for(Map.Entry<String, Method> methodEntry : queries.entrySet()) {
135 					if(consumableText.startsWith(methodEntry.getKey())) {
136 						consumableText = consumableText.substring(methodEntry.getKey().length());
137 						created.addQueryExpression(informer, methodEntry.getValue());
138 						found = true;
139 					}
140 				}
141 				// TODO insert here code for call chaining
142 				if(!found) {
143 					throw new UnableToBuildDueToMissingQueryExpressionException(consumableText, methodString, fieldEntry.getKey(), queries.keySet());
144 				}
145 				for(Combinator c: Combinator.values()) {
146 					if(consumableText.startsWith(c.getText())) {
147 						consumableText = consumableText.substring(c.getText().length());
148 						created.setCombinator(c);
149 					}
150 				}
151 				found = true;
152 			}
153 		}
154 		return consumableText;

159 	 * @param method method to map
160 	 * @return a method resolver for the given method name
161 	 * @category dynamic_method_builder
162 	 */
163 	public MethodResolver createResolver(Method method) throws VirtualMethodCreationException {
164 		Class<?> declaringClass = method.getDeclaringClass();
165 		if(declaringClass.isAssignableFrom(DynamicFinder.class)) {
166 			return new CallMethodOnObjectResolver(backEnd, method);
167 		} else if(declaringClass.isAssignableFrom(DynamicFinderHandler.class)) {
168 			return new CallMethodOnObjectResolver(this, method);
169 		} else if(declaringClass.isAssignableFrom(IdBasedService.class)) {
170 			return new CallMethodOnObjectResolver(backEnd, method);
171 		} else {
172 			return createResolverForDynamicMethod(method);
173 		}
174 	}

176 	/**
177 	 * When using dynamic languages, we can't access to high level type information like in Java, but rather to a name and method parameters (which are actual ones).
178 	 * As a consequence, a weaker type handling is to use
179 	 * @param method
180 	 * @return
181 	 */
182 	private DynamicFinderMethodResolver createResolverForDynamicMethod(String methodName, Object[] methodArgs) {
183 		boolean found = false;
184 		String consumableText = methodName;
185 		
186 		ParametersConstructionMode mode = ParametersConstructionMode.FILTER;
187 		DynamicFinderMethodResolver created = new DynamicFinderMethodResolver(backEnd, methodName, Object.class);
188 		for(Mode m : Mode.values()) {
189 			if(consumableText.startsWith(m.getPrefix())) {
190 				created.setMode(m);
191 				found = true;
192 			}
193 		}
194 		if(!found) {
195 			throw new UnableToBuildDueToMissingModeException(methodName);
196 		}
197 		consumableText = consumableText.substring(created.getMode().getPrefix().length());
198 		Map<String, String> fieldNames = Utils.getUppercasedMap(PropertyProviderUtils.getAllProperties(propertyProvider, backEnd.getContainedClass()));
199 		while(consumableText.length()>0) {
200 			String initialText = consumableText;
201 			consumableText = mode.consumeText(this, methodName, consumableText, fieldNames, created);
202 			if(consumableText.startsWith(SORT_BY_PREFIX)) {
203 				if(mode==ParametersConstructionMode.FILTER) {
204 					mode = ParametersConstructionMode.SORT;
205 					consumableText = consumableText.substring(SORT_BY_PREFIX.length());
206 				}
207 			}
208 			if(initialText.equals(consumableText)) {
209 				throw new UnableToBuildDueToMissingFieldException(consumableText, methodName, fieldNames.keySet());
210 			}
211 		}
212 		return created;
213 	}

216 	 * Create resolver for method object
217 	 * @param method
218 	 * @return
219 	 */
220 	private DynamicFinderMethodResolver createResolverForDynamicMethod(
221 			Method method) {
222 		boolean found = false;
223 		String consumableText = method.getName();
224 		ParametersConstructionMode mode = ParametersConstructionMode.FILTER;
225 		DynamicFinderMethodResolver created = new DynamicFinderMethodResolver(backEnd, method);
226 		for(Mode m : Mode.values()) {
227 			if(consumableText.startsWith(m.getPrefix())) {
228 				created.setMode(m);
229 				found = true;
230 			}
231 		}
232 		if(!found) {
233 			throw new UnableToBuildDueToMissingModeException(method);
234 		}
235 		consumableText = consumableText.substring(created.getMode().getPrefix().length());
236 		Map<String, String> fieldNames = Utils.getUppercasedMap(PropertyProviderUtils.getAllProperties(propertyProvider, backEnd.getContainedClass()));
237 		while(consumableText.length()>0) {
238 			String initialText = consumableText;
239 			consumableText = mode.consumeText(this, method.toGenericString(), consumableText, fieldNames, created);
240 			if(consumableText.startsWith(SORT_BY_PREFIX)) {
241 				if(mode==ParametersConstructionMode.FILTER) {
242 					mode = ParametersConstructionMode.SORT;
243 					consumableText = consumableText.substring(SORT_BY_PREFIX.length());
244 				}
245 			}
246 			if(initialText.equals(consumableText)) {
247 				throw new UnableToBuildDueToMissingFieldException(consumableText, method, fieldNames.keySet());
248 			}
249 		}
250 		// Now created has been terminated, check method corresponds to what we expect
251 		created.checkMethod(method);
252 		return created;
253 	}
