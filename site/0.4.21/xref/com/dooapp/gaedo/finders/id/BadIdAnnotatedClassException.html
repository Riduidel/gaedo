11  /**
12   * That class various id elements does not allow the correct definition of a simple id :
13   * <ul>
14   * <li>Either the class has no field annotated with one of the {@link Annotations.ID} possible values</li>
15   * <li>Or the class has too many elements with such annotations</li>
16   * <li>Or the class has only one field with such annotation, which is good, but the field is not correctly typed</li>
17   * </ul>
18   * As a remainder, for a class to be persisted using GAE, it must have as id field something like
19   * <pre>
20   * 		@Id
21   * 		private long use_any_name_you_want_here;
22   * </pre>
23   * the name id is irrelevant here. However, the annotation and the type are.
24   * @author Nicolas
25   *
26   */
27  public class BadIdAnnotatedClassException extends CrudServiceException {
28  	private static String toString(Collection<Class> expectedIdClasses) {
29  		StringBuilder sOut = new StringBuilder();
30  		if(expectedIdClasses!=null) {
31  			for(Class<?> c : expectedIdClasses) {
32  				sOut.append("\n\t").append(c.getCanonicalName());
33  			}
34  		}
35  		return sOut.toString();
36  	}
37  
38  	public BadIdAnnotatedClassException(Class<?> containedClass, Collection<Class> expectedIdClasses) {
39  		this("Class "+containedClass.getName()+" must have EXACTLY one field annotated with @Id of one of these types : "+toString(expectedIdClasses));
40  	}
41  
42  	protected BadIdAnnotatedClassException(String message) {
43  		super(message);
44  	}
45  
46  	public BadIdAnnotatedClassException(Class<?> containedClass) {
47  		this("Class "+containedClass.getName()+" must have EXACTLY one field annotated with @Id");
48  	}
