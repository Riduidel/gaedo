69  			Index<Edge> edges = indexable.getIndex(Index.EDGES, Edge.class);

72  				GraphBasedPropertyBuilder<DataType> builder = new GraphBasedPropertyBuilder<DataType>(dataClass, driver, namedGraphs);
73  				for(Edge e : edges.get(Properties.label.name(), fieldName)) {
