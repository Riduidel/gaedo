37  public class BluePrintsQueryBuilder<DataType, InformerType extends Informer<DataType>> implements QueryExpressionVisitor {

39  	private AbstractBluePrintsBackedFinderService<?, DataType, InformerType> service;
40  	/**
41  	 * This stack contains only the tests allowing tree building
42  	 */
43  	private Stack<CompoundVertexTest> tests = new Stack<CompoundVertexTest>();
44  
45  	public BluePrintsQueryBuilder(AbstractBluePrintsBackedFinderService<?, DataType, InformerType> service) {
46  		this.service = service;
47  		/* Base test is always a AND one, associated to a test on class (will be used for optimized query building) */
48  		this.tests.push(new AndVertexTest(service.getStrategy(), service.getDriver(), null /* null indicates no property is navigated */));
49  	}
50  
51  	/**
52  	 * Effectively creates the query object that will be used to browse the graph DB
53  	 * @param sortingExpression
54  	 * @return
55  	 */
56  	public GraphExecutableQuery getQuery(SortingExpression sortingExpression) {
57  		// At the end of the visit, there should be only one item in stack : the root one
58  		if(tests.size()!=1) {
59  			throw new InvalidTestStructureException(tests);
60  		}
61  		return new OptimizedGraphExecutableQuery(service, tests.peek(), sortingExpression);
62  //		return new BasicGraphExecutableQuery(service, tests.peek(), sortingExpression);
63  	}
