96  	private static String getNamedInformerForRawType(Map<String, Class> resolvedInformers, Collection<String> qualifiedEnums, Collection<String> imports, String type) {
97  		type = type.trim();
98  		// First try direct qualified name
99  		String returned = getResolvedInformer(resolvedInformers, qualifiedEnums, type);
100 		if(returned==null) {
101 			// Always try qualified names first
102 			for(String s : imports) {
103 				if(returned==null && s.substring(s.lastIndexOf(".")+1).equals(type)) {
104 					returned = getResolvedInformer(resolvedInformers, qualifiedEnums, s);
105 				}
106 			}
107 		}
108 		if(returned==null) {
109 			// Then wildcarded imports
110 			for(String s : imports) {
111 				if(returned==null) {
112 					returned = getResolvedInformer(resolvedInformers, qualifiedEnums, s+"."+type);
113 				}
114 			}
115 		}
116 		if(returned==null) {
117 			returned = type.toString() + INFORMER;
118 			// This is a raw type, so consider finding the raw type (if possible in imports and adding it)
119 			Collection<String> possibleTypes = new LinkedList<String>();
120 			for(String s : imports) {
121 				if(s.endsWith("."+type)) {
122 					possibleTypes.add(s+INFORMER);
123 				}
124 			}
125 			imports.addAll(possibleTypes);
126 		}
127 		// Then give up
128 		return returned;
129 	}

131 	private static String getResolvedInformer(Map<String, Class> resolvedInformers, Collection<String> qualifiedEnums, String s) {
132 		if(qualifiedEnums.contains(s)) {
133 			s = Enum.class.getCanonicalName();
134 		}
135 		// Direct import
136 		if(resolvedInformers.containsKey(s)) {
137 			Class qualifiedClass = resolvedInformers.get(s);
138 			// According to implem, all basic informers are in same package
139 			return BasicFieldInformerLocator.getInformersMapping().get(qualifiedClass).getSimpleName();
140 		}
141 		return null;
142 	}
