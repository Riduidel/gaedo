69  		try {
70  			Class<?> containedClass = Class.forName(containedType);
71  			FinderCrudService service = getServiceRepository().get(containedClass);

80  	/**
81  	 * Create given object from specified arguments
82  	 * @param service
83  	 * @param objectParams
84  	 * @return
85  	 * @throws InstantiationException
86  	 * @throws IllegalAccessException
87  	 * @category CRUD
88  	 */
89  	public Object create(FinderCrudService service,
90  			Map<String, Object> objectParams) throws InstantiationException,
91  			IllegalAccessException {
92  		Map<String, Object> valuesAsTree = (Map<String, Object>) Utils.getValuesAsTree(objectParams).get(RestServiceParams.OBJECT.getPrefix());
93  		Object created = service.getContainedClass().newInstance();
94  		Informer<?> informer = service.getInformer();
95  		for(Object field : informer.getAllFields()) {
96  			if (field instanceof Property) {
97  				Property property = (Property) field;
98  				if(valuesAsTree.containsKey(property.getName())) {
99  					property.set(created, property.fromString(valuesAsTree.get(property.getName()).toString()));
100 				}
101 			}
102 		}
103 		return service.create(created);
104 	}

106 	/**
107 	 * The allmighty get method aggregates three distinct features of gaedo :
108 	 * {@link FinderCrudService#findAll()}, {@link FinderCrudService#find()} and
109 	 * {@link IdBasedService#findById(Object...)}. It does so by providing a
110 	 * rigourous analysis of the parameters given as input to it.
111 	 * 
112 	 * This analysis is split in various steps.
113 	 * <ol>
114 	 * <li>First, the contained type is retrieved (rather easy, since restlet
115 	 * creates the variable for us)</li>
116 	 * <li>Then search properties are looked up. Their lookup is done the
117 	 * following way: They all start with {@value #FILTER_PARAM_PREFIX} and are supposed to be map of
118 	 * properties of input bean. As an example, for the User class of gaedo test
119 	 * beans, "login", "password", "id" and "posts" are valid names. However,
120 	 * there are special cases. If any of {@link FilterAggregator} are used, they generate associated query expressions.</li>
121 	 * <li>In the same way, sort parameters are defined by using {@value #SORT_PARAM_PREFIX} as
122 	 * prefix, and "Ascending" or "Descending" as value.</li>
123 	 * <li>Finally, the search operation is given the same way using the {@value #RETURN_PARAM_PREFIX} parameter</li>
124 	 * </ol>
125 	 * 
126 	 * @return
127 	 * @category REST
128 	 */
129 	@Get
130 	@SuppressWarnings("unchecked") /* Removed since working with typesafe generics is useless here */
131 	public Representation find() {
132 		String containedType = getRequestAttributes().get(
133 				CONTAINED_TYPE_ATTRIBUTE).toString();
134 		try {
135 			Class<?> containedClass = Class.forName(containedType);
136 			FinderCrudService service = getServiceRepository().get(containedClass);
137 			Map<String, Object> filterParams = RestServiceParams.FILTER.getParams(getRequestAttributes());
138 			Map<String, Object> sortParams = RestServiceParams.SORT.getParams(getRequestAttributes());
139 			Map<String, Object> returnParams = RestServiceParams.RETURN.getParams(getRequestAttributes());
140 			Object returnable = find(service, filterParams, sortParams, returnParams);
141 			return represent(returnable);
142 		} catch(Exception e) {
143 			return handleExceptionRestReturn("unable to get data from service associated to "+containedType, e);
144 		}
145 	}

161 	/**
162 	 * Creates a representation from any kind of object.
163 	 * @param source source object to represent
164 	 * @return an XML representation built from a modified XML serialization mechanism
165 	 * @todo replace with use of {@link ConverterService}
166 	 */
167 	private Representation represent(Object source) {
168 		if(source instanceof Serializable) {
169 			return new ObjectRepresentation<Serializable>((Serializable) source);
170 		} else if(source instanceof Iterable) {
171 			// As a matter of fact, collections representation are all limited to MAX_COLLECTION_SIZE
172 			Collection<Object> returned = new LinkedList<Object>();
173 			int index = 0;
174 			for(Object o : (Iterable) source) {
175 				if(index<MAX_COLLECTION_SIZE)
176 					returned.add(o);
177 				else
178 					break;
179 			}
180 			return represent(returned);
181 		} else {
182 			throw new UnrepresentableObjectException(source);
183 		}
184 	}

187 	 * Testable method (for checking all works correctly outside of the restlet box)
188 	 * @param service
189 	 * @param filterParams
190 	 * @param sortParams
191 	 * @param returnParams
192 	 * @return
193 	 * @category CRUD
194 	 */
195 	public Object find(FinderCrudService service,
196 			Map<String, Object> filterParams, Map<String, Object> sortParams,
197 			Map<String, Object> returnParams) {
198 		// TODO find a way to make use of IdBasedService, because that's obviously not the case here
199 		QueryStatement statement =  service.find().matching(queryTranscripter.buildQuery(service, filterParams));
200 		statement = statement.sortBy(sortingTranscripter.buildSorting(service, sortParams));
201 		return returnTranscriptor.buildReturn(statement, returnParams);
202 	}

204 	/**
205 	 * Create an error representation and returns it
206 	 * @param e
207 	 * @param message
208 	 * @return
209 	 */
210 	private AppendableRepresentation buildErrorRepresentation(Exception e,
211 			String message) {
212 		AppendableRepresentation error = new AppendableRepresentation(message);
213 		ByteArrayOutputStream errorStream = new ByteArrayOutputStream();
214 		e.printStackTrace(new PrintStream(errorStream));
215 		try {
216 			error.append("\n\nerror stack\n\n");
217 			error.append(errorStream.toString());
218 		} catch(Exception x) {
219 			logger.log(Level.SEVERE, "unable to append anything to error representation", x);
220 		}
221 		return error;
222 	}
