236 			List<PropertyBagMap> spaceshipList = CollectionUtils.asList(engineeringSpaceships);
237 			assertThat(spaceshipList, IsCollectionContaining.hasItem(ariane1));
238 			// ariane1 is not "valid"
239 			int numberOfvalidSpacesihips = spaceshipList.size()-1;
240 			// now delete it
241 			idService.delete(ariane1);
242 			engineeringSpaceships = propertyBagService.find().matching(new FindByDiscipline(idService.findById(ENGINEERING))).getAll();
243 			spaceshipList = CollectionUtils.asList(engineeringSpaceships);
244 			assertThat(spaceshipList.size(), Is.is(numberOfvalidSpacesihips));
