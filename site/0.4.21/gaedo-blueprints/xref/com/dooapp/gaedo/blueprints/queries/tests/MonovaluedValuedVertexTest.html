41  	 * To match node
42  	 * @param examined

44  	 * @see com.dooapp.gaedo.blueprints.queries.tests.VertexTest#matches(com.tinkerpop.blueprints.pgm.Vertex)
45  	 */
46  	@Override
47  	public boolean matches(Vertex examined) {
48  		// Navigates to the first target edge and perform etest when reached
49  		Vertex currentVertex = examined;
50  		Property finalProperty = null;
51  		// Counting path length allows us to check if we expect a null value
52  		int currentPathLength = 0;
53  		for(Property currentProperty : path) {
54  			Iterator<Edge> edges = strategy.getOutEdgesFor(currentVertex, currentProperty).iterator();
55  			if(edges.hasNext()) {
56  				currentVertex = edges.next().getInVertex();
57  			} else {
58  				return false;
59  			}
60  			finalProperty = currentProperty;
61  			currentPathLength++;
62  		}
63  		// null final property indicates object has no value for that property
64  		if(finalProperty==null) {
65  			return matchesNull();
66  		} else {
67  			return matchesVertex(currentVertex, finalProperty);
68  		}
69  	}
