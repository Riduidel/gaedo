34  	public static <DataType> GraphMappingStrategy<DataType> loadStrategyFor(StrategyType strategy, Class<DataType> containedClass, PropertyProvider propertyProvider, Migrator migrator) {
35  		switch(strategy) {
36  		case beanBased:
37  			return new BeanBasedMappingStrategy<DataType>(containedClass, propertyProvider, migrator);
38  		case graphBased:
39  			return new GraphBasedMappingStrategy<DataType>(containedClass, propertyProvider, migrator);
40  		default:
41  			throw new UnsupportedOperationException("the StrategyType "+strategy.name()+" is not yet supported");
42  		}
43  	}

46  	 * Extract cascade mode of available JPA annotations
47  	 * @param p property to extract cascade annotations from
48  	 * @return the list of cascaded operations
49  	 */
50  	public static Collection<CascadeType> extractCascadeOfJPAAnnotations(Property p) {
51  		Collection<CascadeType> mapping = null;
52  		if (p.getAnnotation(OneToOne.class) != null) {
53  			mapping = GraphUtils.extractCascadeOf(p.getAnnotation(OneToOne.class).cascade());
54  		} else if (p.getAnnotation(OneToMany.class) != null) {
55  			mapping = GraphUtils.extractCascadeOf(p.getAnnotation(OneToMany.class).cascade());
56  		} else if (p.getAnnotation(ManyToMany.class) != null) {
57  			mapping = GraphUtils.extractCascadeOf(p.getAnnotation(ManyToMany.class).cascade());
58  		} else if (p.getAnnotation(ManyToOne.class) != null) {
59  			mapping = GraphUtils.extractCascadeOf(p.getAnnotation(ManyToOne.class).cascade());
60  		} else {
61  			if(!p.hasModifier(Modifier.STATIC) && !p.hasModifier(Modifier.TRANSIENT))
62  				mapping = GraphUtils.extractCascadeOf(new CascadeType[] {CascadeType.ALL});
63  		}
64  		return mapping;
65  	}

79  		Map<Property, Collection<CascadeType>> returned = new TreeMap<Property, Collection<CascadeType>>(new ComparePropertyByNameAndDeclaringClass());
80  		Property[] properties = PropertyProviderUtils.getAllProperties(provider, searchedClass);
81  		for (Property p : properties) {
82  			returned.put(p, extractCascadeOfJPAAnnotations(p));
83  		}
84  		// And, if class is the contained one, add the (potential) Migrator
85  		// property
86  		if (migrator != null) {
87  			// Migrator has no cascade to be done on
88  			returned.put(migrator.getMigratorProperty(returned.keySet()), new LinkedList<CascadeType>());
89  		}
90  		return returned;
