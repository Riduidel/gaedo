33  public class PrevalenceFinderService<DataType extends Serializable, InformerType extends Informer<DataType>>
34  		extends AbstractFinderService<DataType, InformerType> implements
35  		FinderCrudService<DataType, InformerType> {

37  	/**
38  	 * Execution space used to store data
39  	 */
40  	private final ExecutionSpace<String> executionSpace;
41  
42  	private final PropertyChangeEmitter support = new PropertyChangeEmitterImpl();
43  
44  	public PrevalenceFinderService(Class<DataType> containedClass,
45  			Class<InformerType> informerClass,
46  			ProxyBackedInformerFactory factory,
47  			ExecutionSpace<String> executionSpace) {
48  		super(containedClass, informerClass, factory);
49  		// At construction, if none exists, we create a collection linked to class name
50  		this.executionSpace = executionSpace;
51  		if(!executionSpace.execute(new Contains<String>(getStorageName()))) {
52  			executionSpace.execute(new Create<String>(getStorageName(), createDataCollection()));
53  		}
54  	}
55  
56  	/**
57  	 * Get name for storage container
58  	 * @return
59  	 */
60  	private String getStorageName() {
61  		return getContainedClass().getName();
62  	}
63  
64  	/**
65  	 * Create data container (which fortunatly is a collection)
66  	 * @return a {@link HashSet} as it provides good unicity coercion
67  	 */
68  	private Serializable createDataCollection() {
69  		return new HashSet<DataType>();
70  	}
71  
72  	@Override
73  	public DataType create(DataType toCreate) {
74  		executionSpace.execute(new PutInCollection<String>(getStorageName(), toCreate));
75  		return toCreate;
76  	}
77  
78  	@Override
79  	public void delete(DataType toDelete) {
80  		executionSpace.execute(new RemoveFromCollection<String>(getStorageName(), toDelete));
81  	}
82  
83  	@Override
84  	public DataType update(DataType toUpdate) {
85  		return executionSpace.execute(new UpdateInCollection<DataType, String>(getStorageName(), toUpdate));
86  	}
87  
88  	@Override
89  	public Iterable<DataType> findAll() {
90  		return executionSpace.execute(new IterateOverCollection<DataType, String>(getStorageName()));
91  	}
